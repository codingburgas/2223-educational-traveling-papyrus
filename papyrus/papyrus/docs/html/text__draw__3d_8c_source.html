1/*******************************************************************************************
2*
3* raylib [text] example - Draw 3d
4*
5* NOTE: Draw a 2D text in 3D space, each letter is drawn in a quad (or 2 quads if backface is set)
6* where the texture coodinates of each quad map to the texture coordinates of the glyphs
7* inside the font texture.
8*
9* A more efficient approach, i believe, would be to render the text in a render texture and
10* map that texture to a plane and render that, or maybe a shader but my method allows more
11* flexibility...for example to change position of each letter individually to make somethink
12* like a wavy text effect.
13*
14* Special thanks to:
15* @Nighten for the DrawTextStyle() code https://github.com/NightenDushi/Raylib_DrawTextStyle
16* Chris Camacho (codifies - http://bedroomcoders.co.uk/) for the alpha discard shader
17*
18* Example originally created with raylib 3.5, last time updated with raylib 4.0
19*
20* Example contributed by Vlad Adrian (@demizdor) and reviewed by Ramon Santamaria (@raysan5)
21*

451// Draw codepoint at specified position in 3D space
452static void DrawTextCodepoint3D(Font font, int codepoint, Vector3 position, float fontSize, bool backface, Color tint)
453{
454 // Character index position in sprite font
455 // NOTE: In case a codepoint is not available in the font, index returned points to '?'
456 int index = GetGlyphIndex(font, codepoint);
457 float scale = fontSize/(float)font.baseSize;
458
459 // Character destination rectangle on screen
460 // NOTE: We consider charsPadding on drawing
461 position.x += (float)(font.glyphs[index].offsetX - font.glyphPadding)/(float)font.baseSize*scale;
462 position.z += (float)(font.glyphs[index].offsetY - font.glyphPadding)/(float)font.baseSize*scale;
463
464 // Character source rectangle from font texture atlas
465 // NOTE: We consider chars padding when drawing, it could be required for outline/glow shader effects
466 Rectangle srcRec = { font.recs[index].x - (float)font.glyphPadding, font.recs[index].y - (float)font.glyphPadding,
467 font.recs[index].width + 2.0f*font.glyphPadding, font.recs[index].height + 2.0f*font.glyphPadding };
468
469 float width = (float)(font.recs[index].width + 2.0f*font.glyphPadding)/(float)font.baseSize*scale;
470 float height = (float)(font.recs[index].height + 2.0f*font.glyphPadding)/(float)font.baseSize*scale;
471
472 if (font.texture.id > 0)
473 {
474 const float x = 0.0f;
475 const float y = 0.0f;
476 const float z = 0.0f;
477
478 // normalized texture coordinates of the glyph inside the font texture (0.0f -> 1.0f)
479 const float tx = srcRec.x/font.texture.width;
480 const float ty = srcRec.y/font.texture.height;
481 const float tw = (srcRec.x+srcRec.width)/font.texture.width;
482 const float th = (srcRec.y+srcRec.height)/font.texture.height;
483
484 if (SHOW_LETTER_BOUNDRY) DrawCubeWiresV((Vector3){ position.x + width/2, position.y, position.z + height/2}, (Vector3){ width, LETTER_BOUNDRY_SIZE, height }, LETTER_BOUNDRY_COLOR);
485
486 rlCheckRenderBatchLimit(4 + 4*backface);
487 rlSetTexture(font.texture.id);
488
489 rlPushMatrix();
490 rlTranslatef(position.x, position.y, position.z);
491
492 rlBegin(RL_QUADS);
493 rlColor4ub(tint.r, tint.g, tint.b, tint.a);
494
495 // Front Face
496 rlNormal3f(0.0f, 1.0f, 0.0f); // Normal Pointing Up
497 rlTexCoord2f(tx, ty); rlVertex3f(x, y, z); // Top Left Of The Texture and Quad
498 rlTexCoord2f(tx, th); rlVertex3f(x, y, z + height); // Bottom Left Of The Texture and Quad
499 rlTexCoord2f(tw, th); rlVertex3f(x + width, y, z + height); // Bottom Right Of The Texture and Quad
500 rlTexCoord2f(tw, ty); rlVertex3f(x + width, y, z); // Top Right Of The Texture and Quad
501
502 if (backface)
503 {
504 // Back Face
505 rlNormal3f(0.0f, -1.0f, 0.0f); // Normal Pointing Down
506 rlTexCoord2f(tx, ty); rlVertex3f(x, y, z); // Top Right Of The Texture and Quad
507 rlTexCoord2f(tw, ty); rlVertex3f(x + width, y, z); // Top Left Of The Texture and Quad
508 rlTexCoord2f(tw, th); rlVertex3f(x + width, y, z + height); // Bottom Left Of The Texture and Quad
509 rlTexCoord2f(tx, th); rlVertex3f(x, y, z + height); // Bottom Right Of The Texture and Quad
510 }
511 rlEnd();
512 rlPopMatrix();
513
514 rlSetTexture(0);
515 }
516}

518// Draw a 2D text in 3D space
519static void DrawText3D(Font font, const char *text, Vector3 position, float fontSize, float fontSpacing, float lineSpacing, bool backface, Color tint)
520{
521 int length = TextLength(text); // Total length in bytes of the text, scanned by codepoints in loop
522
523 float textOffsetY = 0.0f; // Offset between lines (on line break '\n')
524 float textOffsetX = 0.0f; // Offset X to next character to draw
525
526 float scale = fontSize/(float)font.baseSize;
527
528 for (int i = 0; i < length;)
529 {
530 // Get next codepoint from byte string and glyph index in font
531 int codepointByteCount = 0;
532 int codepoint = GetCodepoint(&text[i], &codepointByteCount);
533 int index = GetGlyphIndex(font, codepoint);
534
535 // NOTE: Normally we exit the decoding sequence as soon as a bad byte is found (and return 0x3f)
536 // but we need to draw all of the bad bytes using the '?' symbol moving one byte
537 if (codepoint == 0x3f) codepointByteCount = 1;
538
539 if (codepoint == '\n')
540 {
541 // NOTE: Fixed line spacing of 1.5 line-height
542 // TODO: Support custom line spacing defined by user
543 textOffsetY += scale + lineSpacing/(float)font.baseSize*scale;
544 textOffsetX = 0.0f;
545 }
546 else
547 {
548 if ((codepoint != ' ') && (codepoint != '\t'))
549 {
550 DrawTextCodepoint3D(font, codepoint, (Vector3){ position.x + textOffsetX, position.y, position.z + textOffsetY }, fontSize, backface, tint);
551 }
552
553 if (font.glyphs[index].advanceX == 0) textOffsetX += (float)(font.recs[index].width + fontSpacing)/(float)font.baseSize*scale;
554 else textOffsetX += (float)(font.glyphs[index].advanceX + fontSpacing)/(float)font.baseSize*scale;
555 }
556
557 i += codepointByteCount; // Move text bytes counter to next codepoint
558 }
559}

561// Measure a text in 3D. For some reason `MeasureTextEx()` just doesn't seem to work so i had to use this instead.
562static Vector3 MeasureText3D(Font font, const char* text, float fontSize, float fontSpacing, float lineSpacing)
563{
564 int len = TextLength(text);
565 int tempLen = 0; // Used to count longer text line num chars
566 int lenCounter = 0;
567
568 float tempTextWidth = 0.0f; // Used to count longer text line width
569
570 float scale = fontSize/(float)font.baseSize;
571 float textHeight = scale;
572 float textWidth = 0.0f;
573
574 int letter = 0; // Current character
575 int index = 0; // Index position in sprite font
576
577 for (int i = 0; i < len; i++)
578 {
579 lenCounter++;
580
581 int next = 0;
582 letter = GetCodepoint(&text[i], &next);
583 index = GetGlyphIndex(font, letter);
584
585 // NOTE: normally we exit the decoding sequence as soon as a bad byte is found (and return 0x3f)
586 // but we need to draw all of the bad bytes using the '?' symbol so to not skip any we set next = 1
587 if (letter == 0x3f) next = 1;
588 i += next - 1;
589
590 if (letter != '\n')
591 {
592 if (font.glyphs[index].advanceX != 0) textWidth += (font.glyphs[index].advanceX+fontSpacing)/(float)font.baseSize*scale;
593 else textWidth += (font.recs[index].width + font.glyphs[index].offsetX)/(float)font.baseSize*scale;
594 }
595 else
596 {
597 if (tempTextWidth < textWidth) tempTextWidth = textWidth;
598 lenCounter = 0;
599 textWidth = 0.0f;
600 textHeight += scale + lineSpacing/(float)font.baseSize*scale;
601 }
602
603 if (tempLen < lenCounter) tempLen = lenCounter;
604 }
605
606 if (tempTextWidth < textWidth) tempTextWidth = textWidth;
607
608 Vector3 vec = { 0 };
609 vec.x = tempTextWidth + (float)((tempLen - 1)*fontSpacing/(float)font.baseSize*scale); // Adds chars spacing to measure
610 vec.y = 0.25f;
611 vec.z = textHeight;
612
613 return vec;
614}

616// Draw a 2D text in 3D space and wave the parts that start with `~~` and end with `~~`.
617// This is a modified version of the original code by @Nighten found here https://github.com/NightenDushi/Raylib_DrawTextStyle
618static void DrawTextWave3D(Font font, const char *text, Vector3 position, float fontSize, float fontSpacing, float lineSpacing, bool backface, WaveTextConfig* config, float time, Color tint)
619{
620 int length = TextLength(text); // Total length in bytes of the text, scanned by codepoints in loop
621
622 float textOffsetY = 0.0f; // Offset between lines (on line break '\n')
623 float textOffsetX = 0.0f; // Offset X to next character to draw
624
625 float scale = fontSize/(float)font.baseSize;
626
627 bool wave = false;
628
629 for (int i = 0, k = 0; i < length; ++k)
630 {
631 // Get next codepoint from byte string and glyph index in font
632 int codepointByteCount = 0;
633 int codepoint = GetCodepoint(&text[i], &codepointByteCount);
634 int index = GetGlyphIndex(font, codepoint);
635
636 // NOTE: Normally we exit the decoding sequence as soon as a bad byte is found (and return 0x3f)
637 // but we need to draw all of the bad bytes using the '?' symbol moving one byte
638 if (codepoint == 0x3f) codepointByteCount = 1;
639
640 if (codepoint == '\n')
641 {
642 // NOTE: Fixed line spacing of 1.5 line-height
643 // TODO: Support custom line spacing defined by user
644 textOffsetY += scale + lineSpacing/(float)font.baseSize*scale;
645 textOffsetX = 0.0f;
646 k = 0;
647 }
648 else if (codepoint == '~')
649 {
650 if (GetCodepoint(&text[i+1], &codepointByteCount) == '~')
651 {
652 codepointByteCount += 1;
653 wave = !wave;
654 }
655 }
656 else
657 {
658 if ((codepoint != ' ') && (codepoint != '\t'))
659 {
660 Vector3 pos = position;
661 if (wave) // Apply the wave effect
662 {
663 pos.x += sinf(time*config->waveSpeed.x-k*config->waveOffset.x)*config->waveRange.x;
664 pos.y += sinf(time*config->waveSpeed.y-k*config->waveOffset.y)*config->waveRange.y;
665 pos.z += sinf(time*config->waveSpeed.z-k*config->waveOffset.z)*config->waveRange.z;
666 }
667
668 DrawTextCodepoint3D(font, codepoint, (Vector3){ pos.x + textOffsetX, pos.y, pos.z + textOffsetY }, fontSize, backface, tint);
669 }
670
671 if (font.glyphs[index].advanceX == 0) textOffsetX += (float)(font.recs[index].width + fontSpacing)/(float)font.baseSize*scale;
672 else textOffsetX += (float)(font.glyphs[index].advanceX + fontSpacing)/(float)font.baseSize*scale;
673 }
674
675 i += codepointByteCount; // Move text bytes counter to next codepoint
676 }
677}
