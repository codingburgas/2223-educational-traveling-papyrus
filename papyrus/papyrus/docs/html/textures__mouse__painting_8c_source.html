16#include "raylib.h"
17
18#define MAX_COLORS_COUNT 23 // Number of colors available
19
20//------------------------------------------------------------------------------------
21// Program main entry point
22//------------------------------------------------------------------------------------
23int main(void)
24{
25 // Initialization
26 //--------------------------------------------------------------------------------------
27 const int screenWidth = 800;
28 const int screenHeight = 450;
29
30 InitWindow(screenWidth, screenHeight, "raylib [textures] example - mouse painting");
31
32 // Colors to choose from
33 Color colors[MAX_COLORS_COUNT] = {
34 RAYWHITE, YELLOW, GOLD, ORANGE, PINK, RED, MAROON, GREEN, LIME, DARKGREEN,
35 SKYBLUE, BLUE, DARKBLUE, PURPLE, VIOLET, DARKPURPLE, BEIGE, BROWN, DARKBROWN,
36 LIGHTGRAY, GRAY, DARKGRAY, BLACK };
37
38 // Define colorsRecs data (for every rectangle)
39 Rectangle colorsRecs[MAX_COLORS_COUNT] = { 0 };
40
41 for (int i = 0; i < MAX_COLORS_COUNT; i++)
42 {
43 colorsRecs[i].x = 10 + 30.0f*i + 2*i;
44 colorsRecs[i].y = 10;
45 colorsRecs[i].width = 30;
46 colorsRecs[i].height = 30;
47 }
48
49 int colorSelected = 0;
50 int colorSelectedPrev = colorSelected;
51 int colorMouseHover = 0;
52 float brushSize = 20.0f;
53 bool mouseWasPressed = false;
54
55 Rectangle btnSaveRec = { 750, 10, 40, 30 };
56 bool btnSaveMouseHover = false;
57 bool showSaveMessage = false;
58 int saveMessageCounter = 0;
59
60 // Create a RenderTexture2D to use as a canvas
61 RenderTexture2D target = LoadRenderTexture(screenWidth, screenHeight);
62
63 // Clear render texture before entering the game loop
64 BeginTextureMode(target);
65 ClearBackground(colors[0]);
66 EndTextureMode();
67
68 SetTargetFPS(120); // Set our game to run at 120 frames-per-second
69 //--------------------------------------------------------------------------------------
70
71 // Main game loop
72 while (!WindowShouldClose()) // Detect window close button or ESC key
73 {
74 // Update
75 //----------------------------------------------------------------------------------
76 Vector2 mousePos = GetMousePosition();
77
78 // Move between colors with keys
79 if (IsKeyPressed(KEY_RIGHT)) colorSelected++;
80 else if (IsKeyPressed(KEY_LEFT)) colorSelected--;
81
82 if (colorSelected >= MAX_COLORS_COUNT) colorSelected = MAX_COLORS_COUNT - 1;
83 else if (colorSelected < 0) colorSelected = 0;
84
85 // Choose color with mouse
86 for (int i = 0; i < MAX_COLORS_COUNT; i++)
87 {
88 if (CheckCollisionPointRec(mousePos, colorsRecs[i]))
89 {
90 colorMouseHover = i;
91 break;
92 }
93 else colorMouseHover = -1;
94 }
95
96 if ((colorMouseHover >= 0) && IsMouseButtonPressed(MOUSE_BUTTON_LEFT))
97 {
98 colorSelected = colorMouseHover;
99 colorSelectedPrev = colorSelected;
100 }
101
102 // Change brush size
103 brushSize += GetMouseWheelMove()*5;
104 if (brushSize < 2) brushSize = 2;
105 if (brushSize > 50) brushSize = 50;
106
107 if (IsKeyPressed(KEY_C))
108 {
109 // Clear render texture to clear color
110 BeginTextureMode(target);
111 ClearBackground(colors[0]);
112 EndTextureMode();
113 }
114
115 if (IsMouseButtonDown(MOUSE_BUTTON_LEFT) || (GetGestureDetected() == GESTURE_DRAG))
116 {
117 // Paint circle into render texture
118 // NOTE: To avoid discontinuous circles, we could store
119 // previous-next mouse points and just draw a line using brush size
120 BeginTextureMode(target);
121 if (mousePos.y > 50) DrawCircle((int)mousePos.x, (int)mousePos.y, brushSize, colors[colorSelected]);
122 EndTextureMode();
123 }
124
125 if (IsMouseButtonDown(MOUSE_BUTTON_RIGHT))
126 {
127 if (!mouseWasPressed)
128 {
129 colorSelectedPrev = colorSelected;
130 colorSelected = 0;
131 }
132
133 mouseWasPressed = true;
134
135 // Erase circle from render texture
136 BeginTextureMode(target);
137 if (mousePos.y > 50) DrawCircle((int)mousePos.x, (int)mousePos.y, brushSize, colors[0]);
138 EndTextureMode();
139 }
140 else if (IsMouseButtonReleased(MOUSE_BUTTON_RIGHT) && mouseWasPressed)
141 {
142 colorSelected = colorSelectedPrev;
143 mouseWasPressed = false;
144 }
145
146 // Check mouse hover save button
147 if (CheckCollisionPointRec(mousePos, btnSaveRec)) btnSaveMouseHover = true;
148 else btnSaveMouseHover = false;
149
150 // Image saving logic
151 // NOTE: Saving painted texture to a default named image
152 if ((btnSaveMouseHover && IsMouseButtonReleased(MOUSE_BUTTON_LEFT)) || IsKeyPressed(KEY_S))
153 {
154 Image image = LoadImageFromTexture(target.texture);
155 ImageFlipVertical(&image);
156 ExportImage(image, "my_amazing_texture_painting.png");
157 UnloadImage(image);
158 showSaveMessage = true;
159 }
160
161 if (showSaveMessage)
162 {
163 // On saving, show a full screen message for 2 seconds
164 saveMessageCounter++;
165 if (saveMessageCounter > 240)
166 {
167 showSaveMessage = false;
168 saveMessageCounter = 0;
169 }
170 }
171 //----------------------------------------------------------------------------------
172
173 // Draw
174 //----------------------------------------------------------------------------------
175 BeginDrawing();
176
177 ClearBackground(RAYWHITE);
178
179 // NOTE: Render texture must be y-flipped due to default OpenGL coordinates (left-bottom)
180 DrawTextureRec(target.texture, (Rectangle) { 0, 0, (float)target.texture.width, (float)-target.texture.height }, (Vector2) { 0, 0 }, WHITE);
181
182 // Draw drawing circle for reference
183 if (mousePos.y > 50)
184 {
185 if (IsMouseButtonDown(MOUSE_BUTTON_RIGHT)) DrawCircleLines((int)mousePos.x, (int)mousePos.y, brushSize, GRAY);
186 else DrawCircle(GetMouseX(), GetMouseY(), brushSize, colors[colorSelected]);
187 }
188
189 // Draw top panel
190 DrawRectangle(0, 0, GetScreenWidth(), 50, RAYWHITE);
191 DrawLine(0, 50, GetScreenWidth(), 50, LIGHTGRAY);
192
193 // Draw color selection rectangles
194 for (int i = 0; i < MAX_COLORS_COUNT; i++) DrawRectangleRec(colorsRecs[i], colors[i]);
195 DrawRectangleLines(10, 10, 30, 30, LIGHTGRAY);
196
197 if (colorMouseHover >= 0) DrawRectangleRec(colorsRecs[colorMouseHover], Fade(WHITE, 0.6f));
198
199 DrawRectangleLinesEx((Rectangle){ colorsRecs[colorSelected].x - 2, colorsRecs[colorSelected].y - 2,
200 colorsRecs[colorSelected].width + 4, colorsRecs[colorSelected].height + 4 }, 2, BLACK);
201
202 // Draw save image button
203 DrawRectangleLinesEx(btnSaveRec, 2, btnSaveMouseHover ? RED : BLACK);
204 DrawText("SAVE!", 755, 20, 10, btnSaveMouseHover ? RED : BLACK);
205
206 // Draw save image message
207 if (showSaveMessage)
208 {
209 DrawRectangle(0, 0, GetScreenWidth(), GetScreenHeight(), Fade(RAYWHITE, 0.8f));
210 DrawRectangle(0, 150, GetScreenWidth(), 80, BLACK);
211 DrawText("IMAGE SAVED: my_amazing_texture_painting.png", 150, 180, 20, RAYWHITE);
212 }
213
214 EndDrawing();
215 //----------------------------------------------------------------------------------
216 }
217
218 // De-Initialization
219 //--------------------------------------------------------------------------------------
220 UnloadRenderTexture(target); // Unload render texture
221
222 CloseWindow(); // Close window and OpenGL context
223 //--------------------------------------------------------------------------------------
224
225 return 0;
226}
