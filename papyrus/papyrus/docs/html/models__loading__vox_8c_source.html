16#include "raylib.h"

18#include "raymath.h" // Required for: MatrixTranslate()

20#define MAX_VOX_FILES 3
21
22//------------------------------------------------------------------------------------
23// Program main entry point
24//------------------------------------------------------------------------------------
25int main(void)
26{
27 // Initialization
28 //--------------------------------------------------------------------------------------
29 const int screenWidth = 800;
30 const int screenHeight = 450;
31
32 const char *voxFileNames[] = {
33 "resources/models/vox/chr_knight.vox",
34 "resources/models/vox/chr_sword.vox",
35 "resources/models/vox/monu9.vox"
36 };
37
38 InitWindow(screenWidth, screenHeight, "raylib [models] example - magicavoxel loading");
39
40 // Define the camera to look into our 3d world
41 Camera camera = { 0 };
42 camera.position = (Vector3){ 10.0f, 10.0f, 10.0f }; // Camera position
43 camera.target = (Vector3){ 0.0f, 0.0f, 0.0f }; // Camera looking at point
44 camera.up = (Vector3){ 0.0f, 1.0f, 0.0f }; // Camera up vector (rotation towards target)
45 camera.fovy = 45.0f; // Camera field-of-view Y
46 camera.projection = CAMERA_PERSPECTIVE; // Camera mode type
47
48 // Load MagicaVoxel files
49 Model models[MAX_VOX_FILES] = { 0 };
50
51 for (int i = 0; i < MAX_VOX_FILES; i++)
52 {
53 // Load VOX file and measure time
54 double t0 = GetTime()*1000.0;
55 models[i] = LoadModel(voxFileNames[i]);
56 double t1 = GetTime()*1000.0;
57
58 TraceLog(LOG_WARNING, TextFormat("[%s] File loaded in %.3f ms", voxFileNames[i], t1 - t0));
59
60 // Compute model translation matrix to center model on draw position (0, 0 , 0)
61 BoundingBox bb = GetModelBoundingBox(models[i]);
62 Vector3 center = { 0 };
63 center.x = bb.min.x + (((bb.max.x - bb.min.x)/2));
64 center.z = bb.min.z + (((bb.max.z - bb.min.z)/2));
65
66 Matrix matTranslate = MatrixTranslate(-center.x, 0, -center.z);
67 models[i].transform = matTranslate;
68 }
69
70 int currentModel = 0;
71
72 SetCameraMode(camera, CAMERA_ORBITAL); // Set a orbital camera mode
73
74 SetTargetFPS(60); // Set our game to run at 60 frames-per-second
75 //--------------------------------------------------------------------------------------
76
77 // Main game loop
78 while (!WindowShouldClose()) // Detect window close button or ESC key
79 {
80 // Update
81 //----------------------------------------------------------------------------------
82 UpdateCamera(&camera);
83
84 // Cycle between models on mouse click
85 if (IsMouseButtonPressed(MOUSE_BUTTON_LEFT)) currentModel = (currentModel + 1)%MAX_VOX_FILES;
86
87 // Cycle between models on key pressed
88 if (IsKeyPressed(KEY_RIGHT))
89 {
90 currentModel++;
91 if (currentModel >= MAX_VOX_FILES) currentModel = 0;
92 }
93 else if (IsKeyPressed(KEY_LEFT))
94 {
95 currentModel--;
96 if (currentModel < 0) currentModel = MAX_VOX_FILES - 1;
97 }
98 //----------------------------------------------------------------------------------
99
100 // Draw
101 //----------------------------------------------------------------------------------
102 BeginDrawing();
103
104 ClearBackground(RAYWHITE);
105
106 // Draw 3D model
107 BeginMode3D(camera);
108
109 DrawModel(models[currentModel], (Vector3){ 0, 0, 0 }, 1.0f, WHITE);
110 DrawGrid(10, 1.0);
111
112 EndMode3D();
113
114 // Display info
115 DrawRectangle(10, 400, 310, 30, Fade(SKYBLUE, 0.5f));
116 DrawRectangleLines(10, 400, 310, 30, Fade(DARKBLUE, 0.5f));
117 DrawText("MOUSE LEFT BUTTON to CYCLE VOX MODELS", 40, 410, 10, BLUE);
118 DrawText(TextFormat("File: %s", GetFileName(voxFileNames[currentModel])), 10, 10, 20, GRAY);
119
120 EndDrawing();
121 //----------------------------------------------------------------------------------
122 }
123
124 // De-Initialization
125 //--------------------------------------------------------------------------------------
126 // Unload models data (GPU VRAM)
127 for (int i = 0; i < MAX_VOX_FILES; i++) UnloadModel(models[i]);
128
129 CloseWindow(); // Close window and OpenGL context
130 //--------------------------------------------------------------------------------------
131
132 return 0;
133}
