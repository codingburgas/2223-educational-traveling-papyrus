37int main(void)
38{
39 // Initialization
40 //--------------------------------------------------------------------------------------
41 const int screenWidth = 800;
42 const int screenHeight = 450;
43
44 SetConfigFlags(FLAG_MSAA_4X_HINT); // Enable Multi Sampling Anti Aliasing 4x (if available)
45 InitWindow(screenWidth, screenHeight, "raylib [shaders] example - fog");
46
47 // Define the camera to look into our 3d world
48 Camera camera = {
49 (Vector3){ 2.0f, 2.0f, 6.0f }, // position
50 (Vector3){ 0.0f, 0.5f, 0.0f }, // target
51 (Vector3){ 0.0f, 1.0f, 0.0f }, // up
52 45.0f, CAMERA_PERSPECTIVE }; // fov, type
53
54 // Load models and texture
55 Model modelA = LoadModelFromMesh(GenMeshTorus(0.4f, 1.0f, 16, 32));
56 Model modelB = LoadModelFromMesh(GenMeshCube(1.0f, 1.0f, 1.0f));
57 Model modelC = LoadModelFromMesh(GenMeshSphere(0.5f, 32, 32));
58 Texture texture = LoadTexture("resources/texel_checker.png");
59
60 // Assign texture to default model material
61 modelA.materials[0].maps[MATERIAL_MAP_DIFFUSE].texture = texture;
62 modelB.materials[0].maps[MATERIAL_MAP_DIFFUSE].texture = texture;
63 modelC.materials[0].maps[MATERIAL_MAP_DIFFUSE].texture = texture;
64
65 // Load shader and set up some uniforms
66 Shader shader = LoadShader(TextFormat("resources/shaders/glsl%i/lighting.vs", GLSL_VERSION),
67 TextFormat("resources/shaders/glsl%i/fog.fs", GLSL_VERSION));
68 shader.locs[SHADER_LOC_MATRIX_MODEL] = GetShaderLocation(shader, "matModel");
69 shader.locs[SHADER_LOC_VECTOR_VIEW] = GetShaderLocation(shader, "viewPos");
70
71 // Ambient light level
72 int ambientLoc = GetShaderLocation(shader, "ambient");
73 SetShaderValue(shader, ambientLoc, (float[4]){ 0.2f, 0.2f, 0.2f, 1.0f }, SHADER_UNIFORM_VEC4);
74
75 float fogDensity = 0.15f;
76 int fogDensityLoc = GetShaderLocation(shader, "fogDensity");
77 SetShaderValue(shader, fogDensityLoc, &fogDensity, SHADER_UNIFORM_FLOAT);
78
79 // NOTE: All models share the same shader
80 modelA.materials[0].shader = shader;
81 modelB.materials[0].shader = shader;
82 modelC.materials[0].shader = shader;
83
84 // Using just 1 point lights
85 CreateLight(LIGHT_POINT, (Vector3){ 0, 2, 6 }, Vector3Zero(), WHITE, shader);
86
87 SetCameraMode(camera, CAMERA_ORBITAL); // Set an orbital camera mode
88
89 SetTargetFPS(60); // Set our game to run at 60 frames-per-second
90 //--------------------------------------------------------------------------------------
91
92 // Main game loop
93 while (!WindowShouldClose()) // Detect window close button or ESC key
94 {
95 // Update
96 //----------------------------------------------------------------------------------
97 UpdateCamera(&camera);
98
99 if (IsKeyDown(KEY_UP))
100 {
101 fogDensity += 0.001f;
102 if (fogDensity > 1.0f) fogDensity = 1.0f;
103 }
104
105 if (IsKeyDown(KEY_DOWN))
106 {
107 fogDensity -= 0.001f;
108 if (fogDensity < 0.0f) fogDensity = 0.0f;
109 }
110
111 SetShaderValue(shader, fogDensityLoc, &fogDensity, SHADER_UNIFORM_FLOAT);
112
113 // Rotate the torus
114 modelA.transform = MatrixMultiply(modelA.transform, MatrixRotateX(-0.025f));
115 modelA.transform = MatrixMultiply(modelA.transform, MatrixRotateZ(0.012f));
116
117 // Update the light shader with the camera view position
118 SetShaderValue(shader, shader.locs[SHADER_LOC_VECTOR_VIEW], &camera.position.x, SHADER_UNIFORM_VEC3);
119 //----------------------------------------------------------------------------------
120
121 // Draw
122 //----------------------------------------------------------------------------------
123 BeginDrawing();
124
125 ClearBackground(GRAY);
126
127 BeginMode3D(camera);
128
129 // Draw the three models
130 DrawModel(modelA, Vector3Zero(), 1.0f, WHITE);
131 DrawModel(modelB, (Vector3){ -2.6f, 0, 0 }, 1.0f, WHITE);
132 DrawModel(modelC, (Vector3){ 2.6f, 0, 0 }, 1.0f, WHITE);
133
134 for (int i = -20; i < 20; i += 2) DrawModel(modelA,(Vector3){ (float)i, 0, 2 }, 1.0f, WHITE);
135
136 EndMode3D();
137
138 DrawText(TextFormat("Use KEY_UP/KEY_DOWN to change fog density [%.2f]", fogDensity), 10, 10, 20, RAYWHITE);
139
140 EndDrawing();
141 //----------------------------------------------------------------------------------
142 }
143
144 // De-Initialization
145 //--------------------------------------------------------------------------------------
146 UnloadModel(modelA); // Unload the model A
147 UnloadModel(modelB); // Unload the model B
148 UnloadModel(modelC); // Unload the model C
149 UnloadTexture(texture); // Unload the texture
150 UnloadShader(shader); // Unload shader
151
152 CloseWindow(); // Close window and OpenGL context
153 //--------------------------------------------------------------------------------------
154
155 return 0;
156}
