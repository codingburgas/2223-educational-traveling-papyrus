3* raylib [shaders] example - OpenGL point particle system
4*
5* Example originally created with raylib 3.8, last time updated with raylib 2.5
6*
7* Example contributed by Stephan Soller (@arkanis) and reviewed by Ramon Santamaria (@raysan5)
8*
9* Example licensed under an unmodified zlib/libpng license, which is an OSI-certified,
10* BSD-like license that allows static linking with closed source software
11*
12* Copyright (c) 2021-2022 Stephan Soller (@arkanis) and Ramon Santamaria (@raysan5)
13*
14********************************************************************************************
15*
16* Mixes raylib and plain OpenGL code to draw a GL_POINTS based particle system. The
17* primary point is to demonstrate raylib and OpenGL interop.
18*
19* rlgl batched draw operations internally so we have to flush the current batch before
20* doing our own OpenGL work (rlDrawRenderBatchActive()).
21*
22* The example also demonstrates how to get the current model view projection matrix of
23* raylib. That way raylib cameras and so on work as expected.
24*
25********************************************************************************************/
26

44#define MAX_PARTICLES 1000
45
46// Particle type
47typedef struct Particle {
48 float x;
49 float y;
50 float period;
51} Particle;
52
53//------------------------------------------------------------------------------------
54// Program main entry point
55//------------------------------------------------------------------------------------
56int main(void)
57{
58 // Initialization
59 //--------------------------------------------------------------------------------------
60 const int screenWidth = 800;
61 const int screenHeight = 450;
62
63 InitWindow(screenWidth, screenHeight, "raylib - point particles");
64
65 Shader shader = LoadShader(TextFormat("resources/shaders/glsl%i/point_particle.vs", GLSL_VERSION),
66 TextFormat("resources/shaders/glsl%i/point_particle.fs", GLSL_VERSION));
67
68 int currentTimeLoc = GetShaderLocation(shader, "currentTime");
69 int colorLoc = GetShaderLocation(shader, "color");
70
71 // Initialize the vertex buffer for the particles and assign each particle random values
72 Particle particles[MAX_PARTICLES] = { 0 };
73
74 for (int i = 0; i < MAX_PARTICLES; i++)
75 {
76 particles[i].x = (float)GetRandomValue(20, screenWidth - 20);
77 particles[i].y = (float)GetRandomValue(50, screenHeight - 20);
78
79 // Give each particle a slightly different period. But don't spread it to much.
80 // This way the particles line up every so often and you get a glimps of what is going on.
81 particles[i].period = (float)GetRandomValue(10, 30)/10.0f;
82 }
83
84 // Create a plain OpenGL vertex buffer with the data and an vertex array object
85 // that feeds the data from the buffer into the vertexPosition shader attribute.
86 GLuint vao = 0;
87 GLuint vbo = 0;
88 glGenVertexArrays(1, &vao);
89 glBindVertexArray(vao);
90 glGenBuffers(1, &vbo);
91 glBindBuffer(GL_ARRAY_BUFFER, vbo);
92 glBufferData(GL_ARRAY_BUFFER, MAX_PARTICLES*sizeof(Particle), particles, GL_STATIC_DRAW);
93 // Note: LoadShader() automatically fetches the attribute index of "vertexPosition" and saves it in shader.locs[SHADER_LOC_VERTEX_POSITION]
94 glVertexAttribPointer(shader.locs[SHADER_LOC_VERTEX_POSITION], 3, GL_FLOAT, GL_FALSE, 0, 0);
95 glEnableVertexAttribArray(0);
96 glBindBuffer(GL_ARRAY_BUFFER, 0);
97 glBindVertexArray(0);
98
99 // Allows the vertex shader to set the point size of each particle individually
100 glEnable(GL_PROGRAM_POINT_SIZE);
101
102 SetTargetFPS(60);
103 //--------------------------------------------------------------------------------------
104
105 // Main game loop
106 while (!WindowShouldClose()) // Detect window close button or ESC key
107 {
108 // Draw
109 //----------------------------------------------------------------------------------
110 BeginDrawing();
111 ClearBackground(WHITE);
112
113 DrawRectangle(10, 10, 210, 30, MAROON);
114 DrawText(TextFormat("%zu particles in one vertex buffer", MAX_PARTICLES), 20, 20, 10, RAYWHITE);
115
116 rlDrawRenderBatchActive(); // Draw iternal buffers data (previous draw calls)
117
118 // Switch to plain OpenGL
119 //------------------------------------------------------------------------------
120 glUseProgram(shader.id);
121
122 glUniform1f(currentTimeLoc, GetTime());
123
124 Vector4 color = ColorNormalize((Color){ 255, 0, 0, 128 });
125 glUniform4fv(colorLoc, 1, (float *)&color);
126
127 // Get the current modelview and projection matrix so the particle system is displayed and transformed
128 Matrix modelViewProjection = MatrixMultiply(rlGetMatrixModelview(), rlGetMatrixProjection());
129
130 glUniformMatrix4fv(shader.locs[SHADER_LOC_MATRIX_MVP], 1, false, MatrixToFloat(modelViewProjection));
131
132 glBindVertexArray(vao);
133 glDrawArrays(GL_POINTS, 0, MAX_PARTICLES);
134 glBindVertexArray(0);
135
136 glUseProgram(0);
137 //------------------------------------------------------------------------------
138
139 DrawFPS(screenWidth - 100, 10);
140
141 EndDrawing();
142 //----------------------------------------------------------------------------------
143 }
144
145 // De-Initialization
146 //--------------------------------------------------------------------------------------
147 glDeleteBuffers(1, &vbo);
148 glDeleteVertexArrays(1, &vao);
149
150 UnloadShader(shader); // Unload shader
151
152 CloseWindow(); // Close window and OpenGL context
153 //--------------------------------------------------------------------------------------
154
155 return 0;
156}
