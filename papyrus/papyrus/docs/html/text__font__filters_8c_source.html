1/*******************************************************************************************
2*
3* raylib [text] example - Font filters
4*
5* NOTE: After font loading, font texture atlas filter could be configured for a softer
6* display of the font when scaling it to different sizes, that way, it's not required
7* to generate multiple fonts at multiple sizes (as long as the scaling is not very different)
8*
9* Example originally created with raylib 1.3, last time updated with raylib 4.2

18#include "raylib.h"
19
20//------------------------------------------------------------------------------------
21// Program main entry point
22//------------------------------------------------------------------------------------
23int main(void)
24{
25 // Initialization
26 //--------------------------------------------------------------------------------------
27 const int screenWidth = 800;
28 const int screenHeight = 450;
29
30 InitWindow(screenWidth, screenHeight, "raylib [text] example - font filters");
31
32 const char msg[50] = "Loaded Font";
33
34 // NOTE: Textures/Fonts MUST be loaded after Window initialization (OpenGL context is required)
35
36 // TTF Font loading with custom generation parameters
37 Font font = LoadFontEx("resources/KAISG.ttf", 96, 0, 0);
38
39 // Generate mipmap levels to use trilinear filtering
40 // NOTE: On 2D drawing it won't be noticeable, it looks like FILTER_BILINEAR
41 GenTextureMipmaps(&font.texture);
42
43 float fontSize = (float)font.baseSize;
44 Vector2 fontPosition = { 40.0f, screenHeight/2.0f - 80.0f };
45 Vector2 textSize = { 0.0f, 0.0f };
46
47 // Setup texture scaling filter
48 SetTextureFilter(font.texture, TEXTURE_FILTER_POINT);
49 int currentFontFilter = 0; // TEXTURE_FILTER_POINT
50
51 SetTargetFPS(60); // Set our game to run at 60 frames-per-second
52 //--------------------------------------------------------------------------------------
53
54 // Main game loop
55 while (!WindowShouldClose()) // Detect window close button or ESC key
56 {
57 // Update
58 //----------------------------------------------------------------------------------
59 fontSize += GetMouseWheelMove()*4.0f;
60
61 // Choose font texture filter method
62 if (IsKeyPressed(KEY_ONE))
63 {
64 SetTextureFilter(font.texture, TEXTURE_FILTER_POINT);
65 currentFontFilter = 0;
66 }
67 else if (IsKeyPressed(KEY_TWO))
68 {
69 SetTextureFilter(font.texture, TEXTURE_FILTER_BILINEAR);
70 currentFontFilter = 1;
71 }
72 else if (IsKeyPressed(KEY_THREE))
73 {
74 // NOTE: Trilinear filter won't be noticed on 2D drawing
75 SetTextureFilter(font.texture, TEXTURE_FILTER_TRILINEAR);
76 currentFontFilter = 2;
77 }
78
79 textSize = MeasureTextEx(font, msg, fontSize, 0);
80
81 if (IsKeyDown(KEY_LEFT)) fontPosition.x -= 10;
82 else if (IsKeyDown(KEY_RIGHT)) fontPosition.x += 10;
83
84 // Load a dropped TTF file dynamically (at current fontSize)
85 if (IsFileDropped())
86 {
87 FilePathList droppedFiles = LoadDroppedFiles();
88
89 // NOTE: We only support first ttf file dropped
90 if (IsFileExtension(droppedFiles.paths[0], ".ttf"))
91 {
92 UnloadFont(font);
93 font = LoadFontEx(droppedFiles.paths[0], (int)fontSize, 0, 0);
94 }
95
96 UnloadDroppedFiles(droppedFiles); // Unload filepaths from memory
97 }
98 //----------------------------------------------------------------------------------
99
100 // Draw
101 //----------------------------------------------------------------------------------
102 BeginDrawing();
103
104 ClearBackground(RAYWHITE);
105
106 DrawText("Use mouse wheel to change font size", 20, 20, 10, GRAY);
107 DrawText("Use KEY_RIGHT and KEY_LEFT to move text", 20, 40, 10, GRAY);
108 DrawText("Use 1, 2, 3 to change texture filter", 20, 60, 10, GRAY);
109 DrawText("Drop a new TTF font for dynamic loading", 20, 80, 10, DARKGRAY);
110
111 DrawTextEx(font, msg, fontPosition, fontSize, 0, BLACK);
112
113 // TODO: It seems texSize measurement is not accurate due to chars offsets...
114 //DrawRectangleLines(fontPosition.x, fontPosition.y, textSize.x, textSize.y, RED);
115
116 DrawRectangle(0, screenHeight - 80, screenWidth, 80, LIGHTGRAY);
117 DrawText(TextFormat("Font size: %02.02f", fontSize), 20, screenHeight - 50, 10, DARKGRAY);
118 DrawText(TextFormat("Text size: [%02.02f, %02.02f]", textSize.x, textSize.y), 20, screenHeight - 30, 10, DARKGRAY);
119 DrawText("CURRENT TEXTURE FILTER:", 250, 400, 20, GRAY);
120
121 if (currentFontFilter == 0) DrawText("POINT", 570, 400, 20, BLACK);
122 else if (currentFontFilter == 1) DrawText("BILINEAR", 570, 400, 20, BLACK);
123 else if (currentFontFilter == 2) DrawText("TRILINEAR", 570, 400, 20, BLACK);
124
125 EndDrawing();
126 //----------------------------------------------------------------------------------
127 }
128
129 // De-Initialization
130 //--------------------------------------------------------------------------------------
131 UnloadFont(font); // Font unloading
132
133 CloseWindow(); // Close window and OpenGL context
134 //--------------------------------------------------------------------------------------
135
136 return 0;
137}
