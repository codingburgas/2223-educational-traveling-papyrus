16#define MAX_INPUT_CHARS 9

18//------------------------------------------------------------------------------------
19// Program main entry point
20//------------------------------------------------------------------------------------
21int main(void)
22{
23 // Initialization
24 //--------------------------------------------------------------------------------------
25 const int screenWidth = 800;
26 const int screenHeight = 450;
27
28 InitWindow(screenWidth, screenHeight, "raylib [text] example - input box");
29
30 char name[MAX_INPUT_CHARS + 1] = "\0"; // NOTE: One extra space required for null terminator char '\0'
31 int letterCount = 0;
32
33 Rectangle textBox = { screenWidth/2.0f - 100, 180, 225, 50 };
34 bool mouseOnText = false;
35
36 int framesCounter = 0;
37
38 SetTargetFPS(10); // Set our game to run at 10 frames-per-second
39 //--------------------------------------------------------------------------------------
40
41 // Main game loop
42 while (!WindowShouldClose()) // Detect window close button or ESC key
43 {
44 // Update
45 //----------------------------------------------------------------------------------
46 if (CheckCollisionPointRec(GetMousePosition(), textBox)) mouseOnText = true;
47 else mouseOnText = false;
48
49 if (mouseOnText)
50 {
51 // Set the window's cursor to the I-Beam
52 SetMouseCursor(MOUSE_CURSOR_IBEAM);
53
54 // Get char pressed (unicode character) on the queue
55 int key = GetCharPressed();
56
57 // Check if more characters have been pressed on the same frame
58 while (key > 0)
59 {
60 // NOTE: Only allow keys in range [32..125]
61 if ((key >= 32) && (key <= 125) && (letterCount < MAX_INPUT_CHARS))
62 {
63 name[letterCount] = (char)key;
64 name[letterCount+1] = '\0'; // Add null terminator at the end of the string.
65 letterCount++;
66 }
67
68 key = GetCharPressed(); // Check next character in the queue
69 }
70
71 if (IsKeyPressed(KEY_BACKSPACE))
72 {
73 letterCount--;
74 if (letterCount < 0) letterCount = 0;
75 name[letterCount] = '\0';
76 }
77 }
78 else SetMouseCursor(MOUSE_CURSOR_DEFAULT);
79
80 if (mouseOnText) framesCounter++;
81 else framesCounter = 0;
82 //----------------------------------------------------------------------------------
83
84 // Draw
85 //----------------------------------------------------------------------------------
86 BeginDrawing();
87
88 ClearBackground(RAYWHITE);
89
90 DrawText("PLACE MOUSE OVER INPUT BOX!", 240, 140, 20, GRAY);
91
92 DrawRectangleRec(textBox, LIGHTGRAY);
93 if (mouseOnText) DrawRectangleLines((int)textBox.x, (int)textBox.y, (int)textBox.width, (int)textBox.height, RED);
94 else DrawRectangleLines((int)textBox.x, (int)textBox.y, (int)textBox.width, (int)textBox.height, DARKGRAY);
95
96 DrawText(name, (int)textBox.x + 5, (int)textBox.y + 8, 40, MAROON);
97
98 DrawText(TextFormat("INPUT CHARS: %i/%i", letterCount, MAX_INPUT_CHARS), 315, 250, 20, DARKGRAY);
99
100 if (mouseOnText)
101 {
102 if (letterCount < MAX_INPUT_CHARS)
103 {
104 // Draw blinking underscore char
105 if (((framesCounter/20)%2) == 0) DrawText("_", (int)textBox.x + 8 + MeasureText(name, 40), (int)textBox.y + 12, 40, MAROON);
106 }
107 else DrawText("Press BACKSPACE to delete chars...", 230, 300, 20, GRAY);
108 }
109
110 EndDrawing();
111 //----------------------------------------------------------------------------------
112 }
113
114 // De-Initialization
115 //--------------------------------------------------------------------------------------
116 CloseWindow(); // Close window and OpenGL context
117 //--------------------------------------------------------------------------------------
118
119 return 0;
120}

124bool IsAnyKeyPressed()

130
131 return keyPressed;
132}
