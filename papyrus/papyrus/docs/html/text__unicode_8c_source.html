311// Fills the emoji array with random emoji (only those emojis present in fontEmoji)
312static void RandomizeEmoji(void)
313{
314 hovered = selected = -1;
315 int start = GetRandomValue(45, 360);
316
317 for (int i = 0; i < SIZEOF(emoji); ++i)
318 {
319 // 0-179 emoji codepoints (from emoji char array) each 4bytes + null char
320 emoji[i].index = GetRandomValue(0, 179)*5;
321
322 // Generate a random color for this emoji
323 emoji[i].color = Fade(ColorFromHSV((float)((start*(i + 1))%360), 0.6f, 0.85f), 0.8f);
324
325 // Set a random message for this emoji
326 emoji[i].message = GetRandomValue(0, SIZEOF(messages) - 1);
327 }

340// Draw text using font inside rectangle limits with support for text selection
341static void DrawTextBoxedSelectable(Font font, const char *text, Rectangle rec, float fontSize, float spacing, bool wordWrap, Color tint, int selectStart, int selectLength, Color selectTint, Color selectBackTint)
342{
343 int length = TextLength(text); // Total length in bytes of the text, scanned by codepoints in loop
344
345 float textOffsetY = 0; // Offset between lines (on line break '\n')
346 float textOffsetX = 0.0f; // Offset X to next character to draw
347
348 float scaleFactor = fontSize/(float)font.baseSize; // Character rectangle scaling factor
349
350 // Word/character wrapping mechanism variables
351 enum { MEASURE_STATE = 0, DRAW_STATE = 1 };
352 int state = wordWrap? MEASURE_STATE : DRAW_STATE;
353
354 int startLine = -1; // Index where to begin drawing (where a line begins)
355 int endLine = -1; // Index where to stop drawing (where a line ends)
356 int lastk = -1; // Holds last value of the character position
357
358 for (int i = 0, k = 0; i < length; i++, k++)
359 {
360 // Get next codepoint from byte string and glyph index in font
361 int codepointByteCount = 0;
362 int codepoint = GetCodepoint(&text[i], &codepointByteCount);
363 int index = GetGlyphIndex(font, codepoint);
364
365 // NOTE: Normally we exit the decoding sequence as soon as a bad byte is found (and return 0x3f)
366 // but we need to draw all of the bad bytes using the '?' symbol moving one byte
367 if (codepoint == 0x3f) codepointByteCount = 1;
368 i += (codepointByteCount - 1);
369
370 float glyphWidth = 0;
371 if (codepoint != '\n')
372 {
373 glyphWidth = (font.glyphs[index].advanceX == 0) ? font.recs[index].width*scaleFactor : font.glyphs[index].advanceX*scaleFactor;
374
375 if (i + 1 < length) glyphWidth = glyphWidth + spacing;
376 }
377
378 // NOTE: When wordWrap is ON we first measure how much of the text we can draw before going outside of the rec container
379 // We store this info in startLine and endLine, then we change states, draw the text between those two variables
380 // and change states again and again recursively until the end of the text (or until we get outside of the container).
381 // When wordWrap is OFF we don't need the measure state so we go to the drawing state immediately
382 // and begin drawing on the next line before we can get outside the container.
383 if (state == MEASURE_STATE)
384 {
385 // TODO: There are multiple types of spaces in UNICODE, maybe it's a good idea to add support for more
386 // Ref: http://jkorpela.fi/chars/spaces.html
387 if ((codepoint == ' ') || (codepoint == '\t') || (codepoint == '\n')) endLine = i;
388
389 if ((textOffsetX + glyphWidth) > rec.width)
390 {
391 endLine = (endLine < 1)? i : endLine;
392 if (i == endLine) endLine -= codepointByteCount;
393 if ((startLine + codepointByteCount) == endLine) endLine = (i - codepointByteCount);
394
395 state = !state;
396 }
397 else if ((i + 1) == length)
398 {
399 endLine = i;
400 state = !state;
401 }
402 else if (codepoint == '\n') state = !state;
403
404 if (state == DRAW_STATE)
405 {
406 textOffsetX = 0;
407 i = startLine;
408 glyphWidth = 0;
409
410 // Save character position when we switch states
411 int tmp = lastk;
412 lastk = k - 1;
413 k = tmp;
414 }
415 }
416 else
417 {
418 if (codepoint == '\n')
419 {
420 if (!wordWrap)
421 {
422 textOffsetY += (font.baseSize + font.baseSize/2)*scaleFactor;
423 textOffsetX = 0;
424 }
425 }
426 else
427 {
428 if (!wordWrap && ((textOffsetX + glyphWidth) > rec.width))
429 {
430 textOffsetY += (font.baseSize + font.baseSize/2)*scaleFactor;
431 textOffsetX = 0;
432 }
433
434 // When text overflows rectangle height limit, just stop drawing
435 if ((textOffsetY + font.baseSize*scaleFactor) > rec.height) break;
436
437 // Draw selection background
438 bool isGlyphSelected = false;
439 if ((selectStart >= 0) && (k >= selectStart) && (k < (selectStart + selectLength)))
440 {
441 DrawRectangleRec((Rectangle){ rec.x + textOffsetX - 1, rec.y + textOffsetY, glyphWidth, (float)font.baseSize*scaleFactor }, selectBackTint);
442 isGlyphSelected = true;
443 }
444
445 // Draw current character glyph
446 if ((codepoint != ' ') && (codepoint != '\t'))
447 {
448 DrawTextCodepoint(font, codepoint, (Vector2){ rec.x + textOffsetX, rec.y + textOffsetY }, fontSize, isGlyphSelected? selectTint : tint);
449 }
450 }
451
452 if (wordWrap && (i == endLine))
453 {
454 textOffsetY += (font.baseSize + font.baseSize/2)*scaleFactor;
455 textOffsetX = 0;
456 startLine = endLine;
457 endLine = -1;
458 glyphWidth = 0;
459 selectStart += lastk - k;
460 k = lastk;
461
462 state = !state;
463 }
464 }
465
466 textOffsetX += glyphWidth;
467 }
