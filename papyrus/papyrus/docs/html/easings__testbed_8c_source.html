99//------------------------------------------------------------------------------------
100// Program main entry point
101//------------------------------------------------------------------------------------
102int main(void)
103{
104 // Initialization
105 //--------------------------------------------------------------------------------------
106 const int screenWidth = 800;
107 const int screenHeight = 450;
108
109 InitWindow(screenWidth, screenHeight, "raylib [easings] example - easings testbed");
110
111 Vector2 ballPosition = { 100.0f, 200.0f };
112
113 float t = 0.0f; // Current time (in any unit measure, but same unit as duration)
114 float d = 300.0f; // Total time it should take to complete (duration)
115 bool paused = true;
116 bool boundedT = true; // If true, t will stop when d >= td, otherwise t will keep adding td to its value every loop
117
118 int easingX = EASING_NONE; // Easing selected for x axis
119 int easingY = EASING_NONE; // Easing selected for y axis
120
121 SetTargetFPS(60);
122 //--------------------------------------------------------------------------------------
123
124 // Main game loop
125 while (!WindowShouldClose()) // Detect window close button or ESC key
126 {
127 // Update
128 //----------------------------------------------------------------------------------
129 if (IsKeyPressed(KEY_T)) boundedT = !boundedT;
130
131 // Choose easing for the X axis
132 if (IsKeyPressed(KEY_RIGHT))
133 {
134 easingX++;
135
136 if (easingX > EASING_NONE) easingX = 0;
137 }
138 else if (IsKeyPressed(KEY_LEFT))
139 {
140 if (easingX == 0) easingX = EASING_NONE;
141 else easingX--;
142 }
143
144 // Choose easing for the Y axis
145 if (IsKeyPressed(KEY_DOWN))
146 {
147 easingY++;
148
149 if (easingY > EASING_NONE) easingY = 0;
150 }
151 else if (IsKeyPressed(KEY_UP))
152 {
153 if (easingY == 0) easingY = EASING_NONE;
154 else easingY--;
155 }
156
157 // Change d (duration) value
158 if (IsKeyPressed(KEY_W) && d < D_MAX - D_STEP) d += D_STEP;
159 else if (IsKeyPressed(KEY_Q) && d > D_MIN + D_STEP) d -= D_STEP;
160
161 if (IsKeyDown(KEY_S) && d < D_MAX - D_STEP_FINE) d += D_STEP_FINE;
162 else if (IsKeyDown(KEY_A) && d > D_MIN + D_STEP_FINE) d -= D_STEP_FINE;
163
164 // Play, pause and restart controls
165 if (IsKeyPressed(KEY_SPACE) || IsKeyPressed(KEY_T) ||
166 IsKeyPressed(KEY_RIGHT) || IsKeyPressed(KEY_LEFT) ||
167 IsKeyPressed(KEY_DOWN) || IsKeyPressed(KEY_UP) ||
168 IsKeyPressed(KEY_W) || IsKeyPressed(KEY_Q) ||
169 IsKeyDown(KEY_S) || IsKeyDown(KEY_A) ||
170 (IsKeyPressed(KEY_ENTER) && (boundedT == true) && (t >= d)))
171 {
172 t = 0.0f;
173 ballPosition.x = 100.0f;
174 ballPosition.y = 100.0f;
175 paused = true;
176 }
177
178 if (IsKeyPressed(KEY_ENTER)) paused = !paused;
179
180 // Movement computation
181 if (!paused && ((boundedT && t < d) || !boundedT))
182 {
183 ballPosition.x = Easings[easingX].func(t, 100.0f, 700.0f - 100.0f, d);
184 ballPosition.y = Easings[easingY].func(t, 100.0f, 400.0f - 100.0f, d);
185 t += 1.0f;
186 }
187 //----------------------------------------------------------------------------------
188
189 // Draw
190 //----------------------------------------------------------------------------------
191 BeginDrawing();
192
193 ClearBackground(RAYWHITE);
194
195 // Draw information text
196 DrawText(TextFormat("Easing x: %s", Easings[easingX].name), 0, FONT_SIZE*2, FONT_SIZE, LIGHTGRAY);
197 DrawText(TextFormat("Easing y: %s", Easings[easingY].name), 0, FONT_SIZE*3, FONT_SIZE, LIGHTGRAY);
198 DrawText(TextFormat("t (%c) = %.2f d = %.2f", (boundedT == true)? 'b' : 'u', t, d), 0, FONT_SIZE*4, FONT_SIZE, LIGHTGRAY);
199
200 // Draw instructions text
201 DrawText("Use ENTER to play or pause movement, use SPACE to restart", 0, GetScreenHeight() - FONT_SIZE*2, FONT_SIZE, LIGHTGRAY);
202 DrawText("Use D and W or A and S keys to change duration", 0, GetScreenHeight() - FONT_SIZE*3, FONT_SIZE, LIGHTGRAY);
203 DrawText("Use LEFT or RIGHT keys to choose easing for the x axis", 0, GetScreenHeight() - FONT_SIZE*4, FONT_SIZE, LIGHTGRAY);
204 DrawText("Use UP or DOWN keys to choose easing for the y axis", 0, GetScreenHeight() - FONT_SIZE*5, FONT_SIZE, LIGHTGRAY);
205
206 // Draw ball
207 DrawCircleV(ballPosition, 16.0f, MAROON);
208
209 EndDrawing();
210 //----------------------------------------------------------------------------------
211 }
212
213 // De-Initialization
214 //--------------------------------------------------------------------------------------
215 CloseWindow();
216 //--------------------------------------------------------------------------------------
217
218 return 0;
219}

222// NoEase function, used when "no easing" is selected for any axis. It just ignores all parameters besides b.
223static float NoEase(float t, float b, float c, float d)
224{
225 float burn = t + b + c + d; // Hack to avoid compiler warning (about unused variables)
226 d += burn;
227
228 return b;
