28//------------------------------------------------------------------------------------
29// Program main entry point
30//------------------------------------------------------------------------------------
31int main(void)
32{
33 // Initialization
34 //--------------------------------------------------------------------------------------
35 const int screenWidth = 800;
36 const int screenHeight = 450;
37
38 InitWindow(screenWidth, screenHeight, "raylib [shaders] example - hot reloading");
39
40 const char *fragShaderFileName = "resources/shaders/glsl%i/reload.fs";
41 long fragShaderFileModTime = GetFileModTime(TextFormat(fragShaderFileName, GLSL_VERSION));
42
43 // Load raymarching shader
44 // NOTE: Defining 0 (NULL) for vertex shader forces usage of internal default vertex shader
45 Shader shader = LoadShader(0, TextFormat(fragShaderFileName, GLSL_VERSION));
46
47 // Get shader locations for required uniforms
48 int resolutionLoc = GetShaderLocation(shader, "resolution");
49 int mouseLoc = GetShaderLocation(shader, "mouse");
50 int timeLoc = GetShaderLocation(shader, "time");
51
52 float resolution[2] = { (float)screenWidth, (float)screenHeight };
53 SetShaderValue(shader, resolutionLoc, resolution, SHADER_UNIFORM_VEC2);
54
55 float totalTime = 0.0f;
56 bool shaderAutoReloading = false;
57
58 SetTargetFPS(60); // Set our game to run at 60 frames-per-second
59 //--------------------------------------------------------------------------------------
60
61 // Main game loop
62 while (!WindowShouldClose()) // Detect window close button or ESC key
63 {
64 // Update
65 //----------------------------------------------------------------------------------
66 totalTime += GetFrameTime();
67 Vector2 mouse = GetMousePosition();
68 float mousePos[2] = { mouse.x, mouse.y };
69
70 // Set shader required uniform values
71 SetShaderValue(shader, timeLoc, &totalTime, SHADER_UNIFORM_FLOAT);
72 SetShaderValue(shader, mouseLoc, mousePos, SHADER_UNIFORM_VEC2);
73
74 // Hot shader reloading
75 if (shaderAutoReloading || (IsMouseButtonPressed(MOUSE_BUTTON_LEFT)))
76 {
77 long currentFragShaderModTime = GetFileModTime(TextFormat(fragShaderFileName, GLSL_VERSION));
78
79 // Check if shader file has been modified
80 if (currentFragShaderModTime != fragShaderFileModTime)
81 {
82 // Try reloading updated shader
83 Shader updatedShader = LoadShader(0, TextFormat(fragShaderFileName, GLSL_VERSION));
84
85 if (updatedShader.id != rlGetShaderIdDefault()) // It was correctly loaded
86 {
87 UnloadShader(shader);
88 shader = updatedShader;
89
90 // Get shader locations for required uniforms
91 resolutionLoc = GetShaderLocation(shader, "resolution");
92 mouseLoc = GetShaderLocation(shader, "mouse");
93 timeLoc = GetShaderLocation(shader, "time");
94
95 // Reset required uniforms
96 SetShaderValue(shader, resolutionLoc, resolution, SHADER_UNIFORM_VEC2);
97 }
98
99 fragShaderFileModTime = currentFragShaderModTime;
100 }
101 }
102
103 if (IsKeyPressed(KEY_A)) shaderAutoReloading = !shaderAutoReloading;
104 //----------------------------------------------------------------------------------
105
106 // Draw
107 //----------------------------------------------------------------------------------
108 BeginDrawing();
109
110 ClearBackground(RAYWHITE);
111
112 // We only draw a white full-screen rectangle, frame is generated in shader
113 BeginShaderMode(shader);
114 DrawRectangle(0, 0, screenWidth, screenHeight, WHITE);
115 EndShaderMode();
116
117 DrawText(TextFormat("PRESS [A] to TOGGLE SHADER AUTOLOADING: %s",
118 shaderAutoReloading? "AUTO" : "MANUAL"), 10, 10, 10, shaderAutoReloading? RED : BLACK);
119 if (!shaderAutoReloading) DrawText("MOUSE CLICK to SHADER RE-LOADING", 10, 30, 10, BLACK);
120
121 DrawText(TextFormat("Shader last modification: %s", asctime(localtime(&fragShaderFileModTime))), 10, 430, 10, BLACK);
122
123 EndDrawing();
124 //----------------------------------------------------------------------------------
125 }
126
127 // De-Initialization
128 //--------------------------------------------------------------------------------------
129 UnloadShader(shader); // Unload shader
130
131 CloseWindow(); // Close window and OpenGL context
132 //--------------------------------------------------------------------------------------
133
134 return 0;
135}
