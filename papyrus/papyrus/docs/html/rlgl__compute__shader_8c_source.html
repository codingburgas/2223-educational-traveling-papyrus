19#include "raylib.h"
20#include "rlgl.h"
21
22#include <stdlib.h>
23
24// IMPORTANT: This must match gol*.glsl GOL_WIDTH constant.
25// This must be a multiple of 16 (check golLogic compute dispatch).
26#define GOL_WIDTH 768
27
28// Maximum amount of queued draw commands (squares draw from mouse down events).
29#define MAX_BUFFERED_TRANSFERTS 48
30
31// Game Of Life Update Command
32typedef struct GolUpdateCmd {
33 unsigned int x; // x coordinate of the gol command
34 unsigned int y; // y coordinate of the gol command
35 unsigned int w; // width of the filled zone
36 unsigned int enabled; // whether to enable or disable zone
37} GolUpdateCmd;
38
39// Game Of Life Update Commands SSBO
40typedef struct GolUpdateSSBO {
41 unsigned int count;
42 GolUpdateCmd commands[MAX_BUFFERED_TRANSFERTS];
43} GolUpdateSSBO;
44
45//------------------------------------------------------------------------------------
46// Program main entry point
47//------------------------------------------------------------------------------------
48int main(void)
49{
50 // Initialization
51 //--------------------------------------------------------------------------------------
52 InitWindow(GOL_WIDTH, GOL_WIDTH, "raylib [rlgl] example - compute shader - game of life");
53
54 const Vector2 resolution = { GOL_WIDTH, GOL_WIDTH };
55 unsigned int brushSize = 8;
56
57 // Game of Life logic compute shader
58 char *golLogicCode = LoadFileText("resources/shaders/glsl430/gol.glsl");
59 unsigned int golLogicShader = rlCompileShader(golLogicCode, RL_COMPUTE_SHADER);
60 unsigned int golLogicProgram = rlLoadComputeShaderProgram(golLogicShader);
61 UnloadFileText(golLogicCode);
62
63 // Game of Life logic render shader
64 Shader golRenderShader = LoadShader(NULL, "resources/shaders/glsl430/gol_render.glsl");
65 int resUniformLoc = GetShaderLocation(golRenderShader, "resolution");
66
67 // Game of Life transfert shader (CPU<->GPU download and upload)
68 char *golTransfertCode = LoadFileText("resources/shaders/glsl430/gol_transfert.glsl");
69 unsigned int golTransfertShader = rlCompileShader(golTransfertCode, RL_COMPUTE_SHADER);
70 unsigned int golTransfertProgram = rlLoadComputeShaderProgram(golTransfertShader);
71 UnloadFileText(golTransfertCode);
72
73 // Load shader storage buffer object (SSBO), id returned
74 unsigned int ssboA = rlLoadShaderBuffer(GOL_WIDTH*GOL_WIDTH*sizeof(unsigned int), NULL, RL_DYNAMIC_COPY);
75 unsigned int ssboB = rlLoadShaderBuffer(GOL_WIDTH*GOL_WIDTH*sizeof(unsigned int), NULL, RL_DYNAMIC_COPY);
76 unsigned int ssboTransfert = rlLoadShaderBuffer(sizeof(GolUpdateSSBO), NULL, RL_DYNAMIC_COPY);
77
78 GolUpdateSSBO transfertBuffer = { 0 };
79
80 // Create a white texture of the size of the window to update
81 // each pixel of the window using the fragment shader: golRenderShader
82 Image whiteImage = GenImageColor(GOL_WIDTH, GOL_WIDTH, WHITE);
83 Texture whiteTex = LoadTextureFromImage(whiteImage);
84 UnloadImage(whiteImage);
85 //--------------------------------------------------------------------------------------
86
87 // Main game loop
88 while (!WindowShouldClose())
89 {
90 // Update
91 //----------------------------------------------------------------------------------
92 brushSize += (int)GetMouseWheelMove();
93
94 if ((IsMouseButtonDown(MOUSE_BUTTON_LEFT) || IsMouseButtonDown(MOUSE_BUTTON_RIGHT))
95 && (transfertBuffer.count < MAX_BUFFERED_TRANSFERTS))
96 {
97 // Buffer a new command
98 transfertBuffer.commands[transfertBuffer.count].x = GetMouseX() - brushSize/2;
99 transfertBuffer.commands[transfertBuffer.count].y = GetMouseY() - brushSize/2;
100 transfertBuffer.commands[transfertBuffer.count].w = brushSize;
101 transfertBuffer.commands[transfertBuffer.count].enabled = IsMouseButtonDown(MOUSE_BUTTON_LEFT);
102 transfertBuffer.count++;
103 }
104 else if (transfertBuffer.count > 0) // Process transfert buffer
105 {
106 // Send SSBO buffer to GPU
107 rlUpdateShaderBuffer(ssboTransfert, &transfertBuffer, sizeof(GolUpdateSSBO), 0);
108
109 // Process SSBO commands on GPU
110 rlEnableShader(golTransfertProgram);
111 rlBindShaderBuffer(ssboA, 1);
112 rlBindShaderBuffer(ssboTransfert, 3);
113 rlComputeShaderDispatch(transfertBuffer.count, 1, 1); // Each GPU unit will process a command!
114 rlDisableShader();
115
116 transfertBuffer.count = 0;
117 }
118 else
119 {
120 // Process game of life logic
121 rlEnableShader(golLogicProgram);
122 rlBindShaderBuffer(ssboA, 1);
123 rlBindShaderBuffer(ssboB, 2);
124 rlComputeShaderDispatch(GOL_WIDTH/16, GOL_WIDTH/16, 1);
125 rlDisableShader();
126
127 // ssboA <-> ssboB
128 int temp = ssboA;
129 ssboA = ssboB;
130 ssboB = temp;
131 }
132
133 rlBindShaderBuffer(ssboA, 1);
134 SetShaderValue(golRenderShader, resUniformLoc, &resolution, SHADER_UNIFORM_VEC2);
135 //----------------------------------------------------------------------------------
136
137 // Draw
138 //----------------------------------------------------------------------------------
139 BeginDrawing();
140
141 ClearBackground(BLANK);
142
143 BeginShaderMode(golRenderShader);
144 DrawTexture(whiteTex, 0, 0, WHITE);
145 EndShaderMode();
146
147 DrawRectangleLines(GetMouseX() - brushSize/2, GetMouseY() - brushSize/2, brushSize, brushSize, RED);
148
149 DrawText("Use Mouse wheel to increase/decrease brush size", 10, 10, 20, WHITE);
150 DrawFPS(GetScreenWidth() - 100, 10);
151
152 EndDrawing();
153 //----------------------------------------------------------------------------------
154 }
155
156 // De-Initialization
157 //--------------------------------------------------------------------------------------
158 // Unload shader buffers objects.
159 rlUnloadShaderBuffer(ssboA);
160 rlUnloadShaderBuffer(ssboB);
161 rlUnloadShaderBuffer(ssboTransfert);
162
163 // Unload compute shader programs
164 rlUnloadShaderProgram(golTransfertProgram);
165 rlUnloadShaderProgram(golLogicProgram);
166
167 UnloadTexture(whiteTex); // Unload white texture
168 UnloadShader(golRenderShader); // Unload rendering fragment shader
169
170 CloseWindow(); // Close window and OpenGL context
171 //--------------------------------------------------------------------------------------
172
173 return 0;
174}
