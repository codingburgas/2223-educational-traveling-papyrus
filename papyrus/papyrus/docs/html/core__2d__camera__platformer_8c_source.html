161void UpdatePlayer(Player *player, EnvItem *envItems, int envItemsLength, float delta)
162{
163 if (IsKeyDown(KEY_LEFT)) player->position.x -= PLAYER_HOR_SPD*delta;
164 if (IsKeyDown(KEY_RIGHT)) player->position.x += PLAYER_HOR_SPD*delta;
165 if (IsKeyDown(KEY_SPACE) && player->canJump)
166 {
167 player->speed = -PLAYER_JUMP_SPD;
168 player->canJump = false;
169 }
170
171 int hitObstacle = 0;
172 for (int i = 0; i < envItemsLength; i++)
173 {
174 EnvItem *ei = envItems + i;
175 Vector2 *p = &(player->position);
176 if (ei->blocking &&
177 ei->rect.x <= p->x &&
178 ei->rect.x + ei->rect.width >= p->x &&
179 ei->rect.y >= p->y &&
180 ei->rect.y <= p->y + player->speed*delta)
181 {
182 hitObstacle = 1;
183 player->speed = 0.0f;
184 p->y = ei->rect.y;
185 }
186 }
187
188 if (!hitObstacle)
189 {
190 player->position.y += player->speed*delta;
191 player->speed += G*delta;
192 player->canJump = false;
193 }
194 else player->canJump = true;
195}

203void UpdateCameraCenterInsideMap(Camera2D *camera, Player *player, EnvItem *envItems, int envItemsLength, float delta, int width, int height)
204{
205 camera->target = player->position;
206 camera->offset = (Vector2){ width/2.0f, height/2.0f };
207 float minX = 1000, minY = 1000, maxX = -1000, maxY = -1000;
208
209 for (int i = 0; i < envItemsLength; i++)
210 {
211 EnvItem *ei = envItems + i;
212 minX = fminf(ei->rect.x, minX);
213 maxX = fmaxf(ei->rect.x + ei->rect.width, maxX);
214 minY = fminf(ei->rect.y, minY);
215 maxY = fmaxf(ei->rect.y + ei->rect.height, maxY);
216 }
217
218 Vector2 max = GetWorldToScreen2D((Vector2){ maxX, maxY }, *camera);
219 Vector2 min = GetWorldToScreen2D((Vector2){ minX, minY }, *camera);
220
221 if (max.x < width) camera->offset.x = width - (max.x - width/2);
222 if (max.y < height) camera->offset.y = height - (max.y - height/2);
223 if (min.x > 0) camera->offset.x = width/2 - min.x;
224 if (min.y > 0) camera->offset.y = height/2 - min.y;
225}

227void UpdateCameraCenterSmoothFollow(Camera2D *camera, Player *player, EnvItem *envItems, int envItemsLength, float delta, int width, int height)
228{
229 static float minSpeed = 30;
230 static float minEffectLength = 10;
231 static float fractionSpeed = 0.8f;
232
233 camera->offset = (Vector2){ width/2.0f, height/2.0f };
234 Vector2 diff = Vector2Subtract(player->position, camera->target);
235 float length = Vector2Length(diff);
236
237 if (length > minEffectLength)
238 {
239 float speed = fmaxf(fractionSpeed*length, minSpeed);
240 camera->target = Vector2Add(camera->target, Vector2Scale(diff, speed*delta/length));
241 }
242}

244void UpdateCameraEvenOutOnLanding(Camera2D *camera, Player *player, EnvItem *envItems, int envItemsLength, float delta, int width, int height)
245{
246 static float evenOutSpeed = 700;
247 static int eveningOut = false;
248 static float evenOutTarget;
249
250 camera->offset = (Vector2){ width/2.0f, height/2.0f };
251 camera->target.x = player->position.x;
252
253 if (eveningOut)
254 {
255 if (evenOutTarget > camera->target.y)
256 {
257 camera->target.y += evenOutSpeed*delta;
258
259 if (camera->target.y > evenOutTarget)
260 {
261 camera->target.y = evenOutTarget;
262 eveningOut = 0;
263 }
264 }
265 else
266 {
267 camera->target.y -= evenOutSpeed*delta;
268
269 if (camera->target.y < evenOutTarget)
270 {
271 camera->target.y = evenOutTarget;
272 eveningOut = 0;
273 }
274 }
275 }
276 else
277 {
278 if (player->canJump && (player->speed == 0) && (player->position.y != camera->target.y))
279 {
280 eveningOut = 1;
281 evenOutTarget = player->position.y;
282 }
283 }
284}

286void UpdateCameraPlayerBoundsPush(Camera2D *camera, Player *player, EnvItem *envItems, int envItemsLength, float delta, int width, int height)
287{
288 static Vector2 bbox = { 0.2f, 0.2f };
289
290 Vector2 bboxWorldMin = GetScreenToWorld2D((Vector2){ (1 - bbox.x)*0.5f*width, (1 - bbox.y)*0.5f*height }, *camera);
291 Vector2 bboxWorldMax = GetScreenToWorld2D((Vector2){ (1 + bbox.x)*0.5f*width, (1 + bbox.y)*0.5f*height }, *camera);
292 camera->offset = (Vector2){ (1 - bbox.x)*0.5f * width, (1 - bbox.y)*0.5f*height };
293
294 if (player->position.x < bboxWorldMin.x) camera->target.x = player->position.x;
295 if (player->position.y < bboxWorldMin.y) camera->target.y = player->position.y;
296 if (player->position.x > bboxWorldMax.x) camera->target.x = bboxWorldMin.x + (player->position.x - bboxWorldMax.x);
297 if (player->position.y > bboxWorldMax.y) camera->target.y = bboxWorldMin.y + (player->position.y - bboxWorldMax.y);
298}
