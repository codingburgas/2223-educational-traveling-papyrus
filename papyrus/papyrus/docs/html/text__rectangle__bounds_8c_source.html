140// Draw text using font inside rectangle limits with support for text selection
141static void DrawTextBoxedSelectable(Font font, const char *text, Rectangle rec, float fontSize, float spacing, bool wordWrap, Color tint, int selectStart, int selectLength, Color selectTint, Color selectBackTint)
142{
143 int length = TextLength(text); // Total length in bytes of the text, scanned by codepoints in loop
144
145 float textOffsetY = 0; // Offset between lines (on line break '\n')
146 float textOffsetX = 0.0f; // Offset X to next character to draw
147
148 float scaleFactor = fontSize/(float)font.baseSize; // Character rectangle scaling factor
149
150 // Word/character wrapping mechanism variables
151 enum { MEASURE_STATE = 0, DRAW_STATE = 1 };
152 int state = wordWrap? MEASURE_STATE : DRAW_STATE;
153
154 int startLine = -1; // Index where to begin drawing (where a line begins)
155 int endLine = -1; // Index where to stop drawing (where a line ends)
156 int lastk = -1; // Holds last value of the character position
157
158 for (int i = 0, k = 0; i < length; i++, k++)
159 {
160 // Get next codepoint from byte string and glyph index in font
161 int codepointByteCount = 0;
162 int codepoint = GetCodepoint(&text[i], &codepointByteCount);
163 int index = GetGlyphIndex(font, codepoint);
164
165 // NOTE: Normally we exit the decoding sequence as soon as a bad byte is found (and return 0x3f)
166 // but we need to draw all of the bad bytes using the '?' symbol moving one byte
167 if (codepoint == 0x3f) codepointByteCount = 1;
168 i += (codepointByteCount - 1);
169
170 float glyphWidth = 0;
171 if (codepoint != '\n')
172 {
173 glyphWidth = (font.glyphs[index].advanceX == 0) ? font.recs[index].width*scaleFactor : font.glyphs[index].advanceX*scaleFactor;
174
175 if (i + 1 < length) glyphWidth = glyphWidth + spacing;
176 }
177
178 // NOTE: When wordWrap is ON we first measure how much of the text we can draw before going outside of the rec container
179 // We store this info in startLine and endLine, then we change states, draw the text between those two variables
180 // and change states again and again recursively until the end of the text (or until we get outside of the container).
181 // When wordWrap is OFF we don't need the measure state so we go to the drawing state immediately
182 // and begin drawing on the next line before we can get outside the container.
183 if (state == MEASURE_STATE)
184 {
185 // TODO: There are multiple types of spaces in UNICODE, maybe it's a good idea to add support for more
186 // Ref: http://jkorpela.fi/chars/spaces.html
187 if ((codepoint == ' ') || (codepoint == '\t') || (codepoint == '\n')) endLine = i;
188
189 if ((textOffsetX + glyphWidth) > rec.width)
190 {
191 endLine = (endLine < 1)? i : endLine;
192 if (i == endLine) endLine -= codepointByteCount;
193 if ((startLine + codepointByteCount) == endLine) endLine = (i - codepointByteCount);
194
195 state = !state;
196 }
197 else if ((i + 1) == length)
198 {
199 endLine = i;
200 state = !state;
201 }
202 else if (codepoint == '\n') state = !state;
203
204 if (state == DRAW_STATE)
205 {
206 textOffsetX = 0;
207 i = startLine;
208 glyphWidth = 0;
209
210 // Save character position when we switch states
211 int tmp = lastk;
212 lastk = k - 1;
213 k = tmp;
214 }
215 }
216 else
217 {
218 if (codepoint == '\n')
219 {
220 if (!wordWrap)
221 {
222 textOffsetY += (font.baseSize + font.baseSize/2)*scaleFactor;
223 textOffsetX = 0;
224 }
225 }
226 else
227 {
228 if (!wordWrap && ((textOffsetX + glyphWidth) > rec.width))
229 {
230 textOffsetY += (font.baseSize + font.baseSize/2)*scaleFactor;
231 textOffsetX = 0;
232 }
233
234 // When text overflows rectangle height limit, just stop drawing
235 if ((textOffsetY + font.baseSize*scaleFactor) > rec.height) break;
236
237 // Draw selection background
238 bool isGlyphSelected = false;
239 if ((selectStart >= 0) && (k >= selectStart) && (k < (selectStart + selectLength)))
240 {
241 DrawRectangleRec((Rectangle){ rec.x + textOffsetX - 1, rec.y + textOffsetY, glyphWidth, (float)font.baseSize*scaleFactor }, selectBackTint);
242 isGlyphSelected = true;
243 }
244
245 // Draw current character glyph
246 if ((codepoint != ' ') && (codepoint != '\t'))
247 {
248 DrawTextCodepoint(font, codepoint, (Vector2){ rec.x + textOffsetX, rec.y + textOffsetY }, fontSize, isGlyphSelected? selectTint : tint);
249 }
250 }
251
252 if (wordWrap && (i == endLine))
253 {
254 textOffsetY += (font.baseSize + font.baseSize/2)*scaleFactor;
255 textOffsetX = 0;
256 startLine = endLine;
257 endLine = -1;
258 glyphWidth = 0;
259 selectStart += lastk - k;
260 k = lastk;
261
262 state = !state;
263 }
264 }
265
266 if ((textOffsetX != 0) || (codepoint != ' ')) textOffsetX += glyphWidth; // avoid leading spaces
267 }
268}
