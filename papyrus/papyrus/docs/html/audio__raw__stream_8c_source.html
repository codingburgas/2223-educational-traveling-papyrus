37// Audio input processing callback
38void AudioInputCallback(void *buffer, unsigned int frames)
39{
40 audioFrequency = frequency + (audioFrequency - frequency)*0.95f;
41 audioFrequency += 1.0f;
42 audioFrequency -= 1.0f;
43 float incr = audioFrequency/44100.0f;
44 short *d = (short *)buffer;
45
46 for (int i = 0; i < frames; i++)
47 {
48 d[i] = (short)(32000.0f*sinf(2*PI*sineIdx));
49 sineIdx += incr;
50 if (sineIdx > 1.0f) sineIdx -= 1.0f;
51 }
52}
