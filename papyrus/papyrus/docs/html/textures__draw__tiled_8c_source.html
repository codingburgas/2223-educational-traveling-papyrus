30{
31 // Initialization
32 //--------------------------------------------------------------------------------------
33 const int screenWidth = 800;
34 const int screenHeight = 450;
35
36 SetConfigFlags(FLAG_WINDOW_RESIZABLE); // Make the window resizable

39 // NOTE: Textures MUST be loaded after Window initialization (OpenGL context is required)
40 Texture texPattern = LoadTexture("resources/patterns.png");
41 SetTextureFilter(texPattern, TEXTURE_FILTER_TRILINEAR); // Makes the texture smoother when upscaled
42
43 // Coordinates for all patterns inside the texture
44 const Rectangle recPattern[] = {
45 (Rectangle){ 3, 3, 66, 66 },
46 (Rectangle){ 75, 3, 100, 100 },
47 (Rectangle){ 3, 75, 66, 66 },
48 (Rectangle){ 7, 156, 50, 50 },
49 (Rectangle){ 85, 106, 90, 45 },
50 (Rectangle){ 75, 154, 100, 60}
51 };
52
53 // Setup colors
54 const Color colors[] = { BLACK, MAROON, ORANGE, BLUE, PURPLE, BEIGE, LIME, RED, DARKGRAY, SKYBLUE };
55 enum { MAX_COLORS = SIZEOF(colors) };
56 Rectangle colorRec[MAX_COLORS] = { 0 };
57
58 // Calculate rectangle for each color
59 for (int i = 0, x = 0, y = 0; i < MAX_COLORS; i++)
60 {
61 colorRec[i].x = 2.0f + MARGIN_SIZE + x;
62 colorRec[i].y = 22.0f + 256.0f + MARGIN_SIZE + y;
63 colorRec[i].width = COLOR_SIZE*2.0f;
64 colorRec[i].height = (float)COLOR_SIZE;
65
66 if (i == (MAX_COLORS/2 - 1))
67 {
68 x = 0;
69 y += COLOR_SIZE + MARGIN_SIZE;
70 }
71 else x += (COLOR_SIZE*2 + MARGIN_SIZE);
72 }
73
74 int activePattern = 0, activeCol = 0;
75 float scale = 1.0f, rotation = 0.0f;
76
77 SetTargetFPS(60);
78 //---------------------------------------------------------------------------------------
79
80 // Main game loop
81 while (!WindowShouldClose()) // Detect window close button or ESC key
82 {
83 // Update
84 //----------------------------------------------------------------------------------
85 // Handle mouse
86 if (IsMouseButtonPressed(MOUSE_BUTTON_LEFT))
87 {
88 const Vector2 mouse = GetMousePosition();
89
90 // Check which pattern was clicked and set it as the active pattern
91 for (int i = 0; i < SIZEOF(recPattern); i++)
92 {
93 if (CheckCollisionPointRec(mouse, (Rectangle){ 2 + MARGIN_SIZE + recPattern[i].x, 40 + MARGIN_SIZE + recPattern[i].y, recPattern[i].width, recPattern[i].height }))
94 {
95 activePattern = i;
96 break;
97 }
98 }
99
100 // Check to see which color was clicked and set it as the active color
101 for (int i = 0; i < MAX_COLORS; ++i)
102 {
103 if (CheckCollisionPointRec(mouse, colorRec[i]))
104 {
105 activeCol = i;
106 break;
107 }
108 }
109 }
110
111 // Handle keys
112
113 // Change scale
114 if (IsKeyPressed(KEY_UP)) scale += 0.25f;
115 if (IsKeyPressed(KEY_DOWN)) scale -= 0.25f;
116 if (scale > 10.0f) scale = 10.0f;
117 else if ( scale <= 0.0f) scale = 0.25f;
118
119 // Change rotation
120 if (IsKeyPressed(KEY_LEFT)) rotation -= 25.0f;
121 if (IsKeyPressed(KEY_RIGHT)) rotation += 25.0f;
122
123 // Reset
124 if (IsKeyPressed(KEY_SPACE)) { rotation = 0.0f; scale = 1.0f; }
125 //----------------------------------------------------------------------------------
126
127 // Draw
128 //----------------------------------------------------------------------------------
129 BeginDrawing();
130 ClearBackground(RAYWHITE);
131
132 // Draw the tiled area
133 DrawTextureTiled(texPattern, recPattern[activePattern], (Rectangle){(float)OPT_WIDTH+MARGIN_SIZE, (float)MARGIN_SIZE, GetScreenWidth() - OPT_WIDTH - 2.0f*MARGIN_SIZE, GetScreenHeight() - 2.0f*MARGIN_SIZE},
134 (Vector2){0.0f, 0.0f}, rotation, scale, colors[activeCol]);
135
136 // Draw options
137 DrawRectangle(MARGIN_SIZE, MARGIN_SIZE, OPT_WIDTH - MARGIN_SIZE, GetScreenHeight() - 2*MARGIN_SIZE, ColorAlpha(LIGHTGRAY, 0.5f));
138
139 DrawText("Select Pattern", 2 + MARGIN_SIZE, 30 + MARGIN_SIZE, 10, BLACK);
140 DrawTexture(texPattern, 2 + MARGIN_SIZE, 40 + MARGIN_SIZE, BLACK);
141 DrawRectangle(2 + MARGIN_SIZE + (int)recPattern[activePattern].x, 40 + MARGIN_SIZE + (int)recPattern[activePattern].y, (int)recPattern[activePattern].width, (int)recPattern[activePattern].height, ColorAlpha(DARKBLUE, 0.3f));
142
143 DrawText("Select Color", 2+MARGIN_SIZE, 10+256+MARGIN_SIZE, 10, BLACK);
144 for (int i = 0; i < MAX_COLORS; i++)
145 {
146 DrawRectangleRec(colorRec[i], colors[i]);
147 if (activeCol == i) DrawRectangleLinesEx(colorRec[i], 3, ColorAlpha(WHITE, 0.5f));
148 }
149
150 DrawText("Scale (UP/DOWN to change)", 2 + MARGIN_SIZE, 80 + 256 + MARGIN_SIZE, 10, BLACK);
151 DrawText(TextFormat("%.2fx", scale), 2 + MARGIN_SIZE, 92 + 256 + MARGIN_SIZE, 20, BLACK);
152
153 DrawText("Rotation (LEFT/RIGHT to change)", 2 + MARGIN_SIZE, 122 + 256 + MARGIN_SIZE, 10, BLACK);
154 DrawText(TextFormat("%.0f degrees", rotation), 2 + MARGIN_SIZE, 134 + 256 + MARGIN_SIZE, 20, BLACK);
155
156 DrawText("Press [SPACE] to reset", 2 + MARGIN_SIZE, 164 + 256 + MARGIN_SIZE, 10, DARKBLUE);
157
158 // Draw FPS
159 DrawText(TextFormat("%i FPS", GetFPS()), 2 + MARGIN_SIZE, 2 + MARGIN_SIZE, 20, BLACK);
160 EndDrawing();
161 //----------------------------------------------------------------------------------
162 }
163
164 // De-Initialization
165 //--------------------------------------------------------------------------------------
166 UnloadTexture(texPattern); // Unload texture
167
168 CloseWindow(); // Close window and OpenGL context
169 //--------------------------------------------------------------------------------------
170
171 return 0;
172}

174// Draw part of a texture (defined by a rectangle) with rotation and scale tiled into dest.
175void DrawTextureTiled(Texture2D texture, Rectangle source, Rectangle dest, Vector2 origin, float rotation, float scale, Color tint)
176{
177 if ((texture.id <= 0) || (scale <= 0.0f)) return; // Wanna see a infinite loop?!...just delete this line!
178 if ((source.width == 0) || (source.height == 0)) return;
179
180 int tileWidth = (int)(source.width*scale), tileHeight = (int)(source.height*scale);
181 if ((dest.width < tileWidth) && (dest.height < tileHeight))
182 {
183 // Can fit only one tile
184 DrawTexturePro(texture, (Rectangle){source.x, source.y, ((float)dest.width/tileWidth)*source.width, ((float)dest.height/tileHeight)*source.height},
185 (Rectangle){dest.x, dest.y, dest.width, dest.height}, origin, rotation, tint);
186 }
187 else if (dest.width <= tileWidth)
188 {
189 // Tiled vertically (one column)
190 int dy = 0;
191 for (;dy+tileHeight < dest.height; dy += tileHeight)
192 {
193 DrawTexturePro(texture, (Rectangle){source.x, source.y, ((float)dest.width/tileWidth)*source.width, source.height}, (Rectangle){dest.x, dest.y + dy, dest.width, (float)tileHeight}, origin, rotation, tint);
194 }
195
196 // Fit last tile
197 if (dy < dest.height)
198 {
199 DrawTexturePro(texture, (Rectangle){source.x, source.y, ((float)dest.width/tileWidth)*source.width, ((float)(dest.height - dy)/tileHeight)*source.height},
200 (Rectangle){dest.x, dest.y + dy, dest.width, dest.height - dy}, origin, rotation, tint);
201 }
202 }
203 else if (dest.height <= tileHeight)
204 {
205 // Tiled horizontally (one row)
206 int dx = 0;
207 for (;dx+tileWidth < dest.width; dx += tileWidth)
208 {
209 DrawTexturePro(texture, (Rectangle){source.x, source.y, source.width, ((float)dest.height/tileHeight)*source.height}, (Rectangle){dest.x + dx, dest.y, (float)tileWidth, dest.height}, origin, rotation, tint);
210 }
211
212 // Fit last tile
213 if (dx < dest.width)
214 {
215 DrawTexturePro(texture, (Rectangle){source.x, source.y, ((float)(dest.width - dx)/tileWidth)*source.width, ((float)dest.height/tileHeight)*source.height},
216 (Rectangle){dest.x + dx, dest.y, dest.width - dx, dest.height}, origin, rotation, tint);
217 }
218 }
219 else
220 {
221 // Tiled both horizontally and vertically (rows and columns)
222 int dx = 0;
223 for (;dx+tileWidth < dest.width; dx += tileWidth)
224 {
225 int dy = 0;
226 for (;dy+tileHeight < dest.height; dy += tileHeight)
227 {
228 DrawTexturePro(texture, source, (Rectangle){dest.x + dx, dest.y + dy, (float)tileWidth, (float)tileHeight}, origin, rotation, tint);
229 }
230
231 if (dy < dest.height)
232 {
233 DrawTexturePro(texture, (Rectangle){source.x, source.y, source.width, ((float)(dest.height - dy)/tileHeight)*source.height},
234 (Rectangle){dest.x + dx, dest.y + dy, (float)tileWidth, dest.height - dy}, origin, rotation, tint);
235 }
236 }
237
238 // Fit last column of tiles
239 if (dx < dest.width)
240 {
241 int dy = 0;
242 for (;dy+tileHeight < dest.height; dy += tileHeight)
243 {
244 DrawTexturePro(texture, (Rectangle){source.x, source.y, ((float)(dest.width - dx)/tileWidth)*source.width, source.height},
245 (Rectangle){dest.x + dx, dest.y + dy, dest.width - dx, (float)tileHeight}, origin, rotation, tint);
246 }
247
248 // Draw final tile in the bottom right corner
249 if (dy < dest.height)
250 {
251 DrawTexturePro(texture, (Rectangle){source.x, source.y, ((float)(dest.width - dx)/tileWidth)*source.width, ((float)(dest.height - dy)/tileHeight)*source.height},
252 (Rectangle){dest.x + dx, dest.y + dy, dest.width - dx, dest.height - dy}, origin, rotation, tint);
253 }
254 }
255 }
