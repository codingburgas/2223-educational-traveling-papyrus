22//------------------------------------------------------------------------------------
23// Program main entry point
24//------------------------------------------------------------------------------------
25int main(void)
26{
27 // Initialization
28 //--------------------------------------------------------------------------------------
29 const int screenWidth = 800;
30 const int screenHeight = 450;
31
32 InitWindow(screenWidth, screenHeight, "raylib [models] example - M3D model loading");
33
34 // Define the camera to look into our 3d world
35 Camera camera = { 0 };
36 camera.position = (Vector3){ 1.5f, 1.5f, 1.5f }; // Camera position
37 camera.target = (Vector3){ 0.0f, 0.4f, 0.0f }; // Camera looking at point
38 camera.up = (Vector3){ 0.0f, 1.0f, 0.0f }; // Camera up vector (rotation towards target)
39 camera.fovy = 45.0f; // Camera field-of-view Y
40 camera.projection = CAMERA_PERSPECTIVE; // Camera mode type
41 Vector3 position = { 0.0f, 0.0f, 0.0f }; // Set model position
42
43 char modelFileName[128] = "resources/models/m3d/CesiumMan.m3d";
44 bool drawMesh = 1;
45 bool drawSkeleton = 1;
46 bool animPlaying = false; // Store anim state, what to draw
47
48 // Load model
49 Model model = LoadModel(modelFileName); // Load the bind-pose model mesh and basic data
50
51 // Load animations
52 unsigned int animsCount = 0;
53 int animFrameCounter = 0, animId = 0;
54 ModelAnimation *anims = LoadModelAnimations(modelFileName, &animsCount); // Load skeletal animation data
55
56 SetCameraMode(camera, CAMERA_FREE); // Set free camera mode
57
58 SetTargetFPS(60); // Set our game to run at 60 frames-per-second
59 //--------------------------------------------------------------------------------------
60
61 // Main game loop
62 while (!WindowShouldClose()) // Detect window close button or ESC key
63 {
64 // Update
65 //----------------------------------------------------------------------------------
66 UpdateCamera(&camera);
67
68 if (animsCount)
69 {
70 // Play animation when spacebar is held down (or step one frame with N)
71 if (IsKeyDown(KEY_SPACE) || IsKeyPressed(KEY_N))
72 {
73 animFrameCounter++;
74
75 if (animFrameCounter >= anims[animId].frameCount) animFrameCounter = 0;
76
77 UpdateModelAnimation(model, anims[animId], animFrameCounter);
78 animPlaying = true;
79 }
80
81 // Select animation by pressing A
82 if (IsKeyPressed(KEY_A))
83 {
84 animFrameCounter = 0;
85 animId++;
86
87 if (animId >= animsCount) animId = 0;
88 UpdateModelAnimation(model, anims[animId], 0);
89 animPlaying = true;
90 }
91 }
92
93 // Toggle skeleton drawing
94 if (IsKeyPressed(KEY_S)) drawSkeleton ^= 1;
95
96 // Toggle mesh drawing
97 if (IsKeyPressed(KEY_M)) drawMesh ^= 1;
98 //----------------------------------------------------------------------------------
99
100 // Draw
101 //----------------------------------------------------------------------------------
102 BeginDrawing();
103
104 ClearBackground(RAYWHITE);
105
106 BeginMode3D(camera);
107
108 // Draw 3d model with texture
109 if (drawMesh) DrawModel(model, position, 1.0f, WHITE);
110
111 // Draw the animated skeleton
112 if (drawSkeleton)
113 {
114 // Loop to (boneCount - 1) because the last one is a special "no bone" bone,
115 // needed to workaround buggy models
116 // without a -1, we would always draw a cube at the origin
117 for (int i = 0; i < model.boneCount - 1; i++)
118 {
119 // By default the model is loaded in bind-pose by LoadModel().
120 // But if UpdateModelAnimation() has been called at least once
121 // then the model is already in animation pose, so we need the animated skeleton
122 if (!animPlaying || !animsCount)
123 {
124 // Display the bind-pose skeleton
125 DrawCube(model.bindPose[i].translation, 0.04f, 0.04f, 0.04f, RED);
126
127 if (model.bones[i].parent >= 0)
128 {
129 DrawLine3D(model.bindPose[i].translation,
130 model.bindPose[model.bones[i].parent].translation, RED);
131 }
132 }
133 else
134 {
135 // Display the frame-pose skeleton
136 DrawCube(anims[animId].framePoses[animFrameCounter][i].translation, 0.05f, 0.05f, 0.05f, RED);
137
138 if (anims[animId].bones[i].parent >= 0)
139 {
140 DrawLine3D(anims[animId].framePoses[animFrameCounter][i].translation,
141 anims[animId].framePoses[animFrameCounter][anims[animId].bones[i].parent].translation, RED);
142 }
143 }
144 }
145 }
146
147 DrawGrid(10, 1.0f); // Draw a grid
148
149 EndMode3D();
150
151 DrawText("PRESS SPACE to PLAY MODEL ANIMATION", 10, GetScreenHeight() - 60, 10, MAROON);
152 DrawText("PRESS A to CYCLE THROUGH ANIMATIONS", 10, GetScreenHeight() - 40, 10, DARKGRAY);
153 DrawText("PRESS M to toggle MESH, S to toggle SKELETON DRAWING", 10, GetScreenHeight() - 20, 10, DARKGRAY);
154 DrawText("(c) CesiumMan model by KhronosGroup", GetScreenWidth() - 210, GetScreenHeight() - 20, 10, GRAY);
155
156 EndDrawing();
157 //----------------------------------------------------------------------------------
158 }
159
160 // De-Initialization
161 //--------------------------------------------------------------------------------------
162
163 // Unload model animations data
164 UnloadModelAnimations(anims, animsCount);
165
166 UnloadModel(model); // Unload model
167
168 CloseWindow(); // Close window and OpenGL context
169 //--------------------------------------------------------------------------------------
170
171 return 0;
172}
