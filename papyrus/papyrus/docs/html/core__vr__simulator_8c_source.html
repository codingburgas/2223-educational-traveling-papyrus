25int main(void)

27 // Initialization
28 //--------------------------------------------------------------------------------------
29 const int screenWidth = 800;
30 const int screenHeight = 450;
31
32 // NOTE: screenWidth/screenHeight should match VR device aspect ratio
33 InitWindow(screenWidth, screenHeight, "raylib [core] example - vr simulator");
34
35 // VR device parameters definition
36 VrDeviceInfo device = {
37 // Oculus Rift CV1 parameters for simulator
38 .hResolution = 2160, // Horizontal resolution in pixels
39 .vResolution = 1200, // Vertical resolution in pixels
40 .hScreenSize = 0.133793f, // Horizontal size in meters
41 .vScreenSize = 0.0669f, // Vertical size in meters
42 .vScreenCenter = 0.04678f, // Screen center in meters
43 .eyeToScreenDistance = 0.041f, // Distance between eye and display in meters
44 .lensSeparationDistance = 0.07f, // Lens separation distance in meters
45 .interpupillaryDistance = 0.07f, // IPD (distance between pupils) in meters
46
47 // NOTE: CV1 uses fresnel-hybrid-asymmetric lenses with specific compute shaders
48 // Following parameters are just an approximation to CV1 distortion stereo rendering
49 .lensDistortionValues[0] = 1.0f, // Lens distortion constant parameter 0
50 .lensDistortionValues[1] = 0.22f, // Lens distortion constant parameter 1
51 .lensDistortionValues[2] = 0.24f, // Lens distortion constant parameter 2
52 .lensDistortionValues[3] = 0.0f, // Lens distortion constant parameter 3
53 .chromaAbCorrection[0] = 0.996f, // Chromatic aberration correction parameter 0
54 .chromaAbCorrection[1] = -0.004f, // Chromatic aberration correction parameter 1
55 .chromaAbCorrection[2] = 1.014f, // Chromatic aberration correction parameter 2
56 .chromaAbCorrection[3] = 0.0f, // Chromatic aberration correction parameter 3
57 };
58
59 // Load VR stereo config for VR device parameteres (Oculus Rift CV1 parameters)
60 VrStereoConfig config = LoadVrStereoConfig(device);
61
62 // Distortion shader (uses device lens distortion and chroma)
63 Shader distortion = LoadShader(0, TextFormat("resources/distortion%i.fs", GLSL_VERSION));
64
65 // Update distortion shader with lens and distortion-scale parameters
66 SetShaderValue(distortion, GetShaderLocation(distortion, "leftLensCenter"),
67 config.leftLensCenter, SHADER_UNIFORM_VEC2);
68 SetShaderValue(distortion, GetShaderLocation(distortion, "rightLensCenter"),
69 config.rightLensCenter, SHADER_UNIFORM_VEC2);
70 SetShaderValue(distortion, GetShaderLocation(distortion, "leftScreenCenter"),
71 config.leftScreenCenter, SHADER_UNIFORM_VEC2);
72 SetShaderValue(distortion, GetShaderLocation(distortion, "rightScreenCenter"),
73 config.rightScreenCenter, SHADER_UNIFORM_VEC2);
74
75 SetShaderValue(distortion, GetShaderLocation(distortion, "scale"),
76 config.scale, SHADER_UNIFORM_VEC2);
77 SetShaderValue(distortion, GetShaderLocation(distortion, "scaleIn"),
78 config.scaleIn, SHADER_UNIFORM_VEC2);
79 SetShaderValue(distortion, GetShaderLocation(distortion, "deviceWarpParam"),
80 device.lensDistortionValues, SHADER_UNIFORM_VEC4);
81 SetShaderValue(distortion, GetShaderLocation(distortion, "chromaAbParam"),
82 device.chromaAbCorrection, SHADER_UNIFORM_VEC4);
83
84 // Initialize framebuffer for stereo rendering
85 // NOTE: Screen size should match HMD aspect ratio
86 RenderTexture2D target = LoadRenderTexture(device.hResolution, device.vResolution);
87
88 // The target's height is flipped (in the source Rectangle), due to OpenGL reasons
89 Rectangle sourceRec = { 0.0f, 0.0f, (float)target.texture.width, -(float)target.texture.height };
90 Rectangle destRec = { 0.0f, 0.0f, (float)GetScreenWidth(), (float)GetScreenHeight() };
91
92 // Define the camera to look into our 3d world
93 Camera camera = { 0 };
94 camera.position = (Vector3){ 5.0f, 2.0f, 5.0f }; // Camera position
95 camera.target = (Vector3){ 0.0f, 2.0f, 0.0f }; // Camera looking at point
96 camera.up = (Vector3){ 0.0f, 1.0f, 0.0f }; // Camera up vector
97 camera.fovy = 60.0f; // Camera field-of-view Y
98 camera.projection = CAMERA_PERSPECTIVE; // Camera type
99
100 Vector3 cubePosition = { 0.0f, 0.0f, 0.0f };
101
102 SetCameraMode(camera, CAMERA_FIRST_PERSON); // Set first person camera mode
103
104 SetTargetFPS(90); // Set our game to run at 90 frames-per-second
105 //--------------------------------------------------------------------------------------
106
107 // Main game loop
108 while (!WindowShouldClose()) // Detect window close button or ESC key
109 {
110 // Update
111 //----------------------------------------------------------------------------------
112 UpdateCamera(&camera);
113 //----------------------------------------------------------------------------------
114
115 // Draw
116 //----------------------------------------------------------------------------------
117 BeginTextureMode(target);
118 ClearBackground(RAYWHITE);
119 BeginVrStereoMode(config);
120 BeginMode3D(camera);
121
122 DrawCube(cubePosition, 2.0f, 2.0f, 2.0f, RED);
123 DrawCubeWires(cubePosition, 2.0f, 2.0f, 2.0f, MAROON);
124 DrawGrid(40, 1.0f);
125
126 EndMode3D();
127 EndVrStereoMode();
128 EndTextureMode();
129
130 BeginDrawing();
131 ClearBackground(RAYWHITE);
132 BeginShaderMode(distortion);
133 DrawTexturePro(target.texture, sourceRec, destRec, (Vector2){ 0.0f, 0.0f }, 0.0f, WHITE);
134 EndShaderMode();
135 DrawFPS(10, 10);
136 EndDrawing();
137 //----------------------------------------------------------------------------------
138 }
139
140 // De-Initialization
141 //--------------------------------------------------------------------------------------
142 UnloadVrStereoConfig(config); // Unload stereo config
143
144 UnloadRenderTexture(target); // Unload stereo render fbo
145 UnloadShader(distortion); // Unload distortion shader
146
147 CloseWindow(); // Close window and OpenGL context
148 //--------------------------------------------------------------------------------------
149
150 return 0;
151}
