21 // Initialization

23 const int screenWidth = 800;
24 const int screenHeight = 450;
25
26 // Possible window flags
27 /*
28 FLAG_VSYNC_HINT
29 FLAG_FULLSCREEN_MODE -> not working properly -> wrong scaling!
30 FLAG_WINDOW_RESIZABLE
31 FLAG_WINDOW_UNDECORATED
32 FLAG_WINDOW_TRANSPARENT
33 FLAG_WINDOW_HIDDEN
34 FLAG_WINDOW_MINIMIZED -> Not supported on window creation
35 FLAG_WINDOW_MAXIMIZED -> Not supported on window creation
36 FLAG_WINDOW_UNFOCUSED
37 FLAG_WINDOW_TOPMOST
38 FLAG_WINDOW_HIGHDPI -> errors after minimize-resize, fb size is recalculated
39 FLAG_WINDOW_ALWAYS_RUN
40 FLAG_MSAA_4X_HINT
41 */
42
43 // Set configuration flags for window creation
44 //SetConfigFlags(FLAG_VSYNC_HINT | FLAG_MSAA_4X_HINT | FLAG_WINDOW_HIGHDPI);
45 InitWindow(screenWidth, screenHeight, "raylib [core] example - window flags");
46
47 Vector2 ballPosition = { GetScreenWidth() / 2.0f, GetScreenHeight() / 2.0f };
48 Vector2 ballSpeed = { 5.0f, 4.0f };
49 float ballRadius = 20;
50
51 int framesCounter = 0;
52
53 //SetTargetFPS(60); // Set our game to run at 60 frames-per-second
54 //----------------------------------------------------------
55
56 // Main game loop
57 while (!WindowShouldClose()) // Detect window close button or ESC key
58 {
59 // Update
60 //-----------------------------------------------------
61 if (IsKeyPressed(KEY_F)) ToggleFullscreen(); // modifies window size when scaling!
62
63 if (IsKeyPressed(KEY_R))
64 {
65 if (IsWindowState(FLAG_WINDOW_RESIZABLE)) ClearWindowState(FLAG_WINDOW_RESIZABLE);
66 else SetWindowState(FLAG_WINDOW_RESIZABLE);
67 }
68
69 if (IsKeyPressed(KEY_D))
70 {
71 if (IsWindowState(FLAG_WINDOW_UNDECORATED)) ClearWindowState(FLAG_WINDOW_UNDECORATED);
72 else SetWindowState(FLAG_WINDOW_UNDECORATED);
73 }
74
75 if (IsKeyPressed(KEY_H))
76 {
77 if (!IsWindowState(FLAG_WINDOW_HIDDEN)) SetWindowState(FLAG_WINDOW_HIDDEN);
78
79 framesCounter = 0;
80 }
81
82 if (IsWindowState(FLAG_WINDOW_HIDDEN))
83 {
84 framesCounter++;
85 if (framesCounter >= 240) ClearWindowState(FLAG_WINDOW_HIDDEN); // Show window after 3 seconds
86 }
87
88 if (IsKeyPressed(KEY_N))
89 {
90 if (!IsWindowState(FLAG_WINDOW_MINIMIZED)) MinimizeWindow();
91
92 framesCounter = 0;
93 }
94
95 if (IsWindowState(FLAG_WINDOW_MINIMIZED))
96 {
97 framesCounter++;
98 if (framesCounter >= 240) RestoreWindow(); // Restore window after 3 seconds
99 }
100
101 if (IsKeyPressed(KEY_M))
102 {
103 // NOTE: Requires FLAG_WINDOW_RESIZABLE enabled!
104 if (IsWindowState(FLAG_WINDOW_MAXIMIZED)) RestoreWindow();
105 else MaximizeWindow();
106 }
107
108 if (IsKeyPressed(KEY_U))
109 {
110 if (IsWindowState(FLAG_WINDOW_UNFOCUSED)) ClearWindowState(FLAG_WINDOW_UNFOCUSED);
111 else SetWindowState(FLAG_WINDOW_UNFOCUSED);
112 }
113
114 if (IsKeyPressed(KEY_T))
115 {
116 if (IsWindowState(FLAG_WINDOW_TOPMOST)) ClearWindowState(FLAG_WINDOW_TOPMOST);
117 else SetWindowState(FLAG_WINDOW_TOPMOST);
118 }
119
120 if (IsKeyPressed(KEY_A))
121 {
122 if (IsWindowState(FLAG_WINDOW_ALWAYS_RUN)) ClearWindowState(FLAG_WINDOW_ALWAYS_RUN);
123 else SetWindowState(FLAG_WINDOW_ALWAYS_RUN);
124 }
125
126 if (IsKeyPressed(KEY_V))
127 {
128 if (IsWindowState(FLAG_VSYNC_HINT)) ClearWindowState(FLAG_VSYNC_HINT);
129 else SetWindowState(FLAG_VSYNC_HINT);
130 }
131
132 // Bouncing ball logic
133 ballPosition.x += ballSpeed.x;
134 ballPosition.y += ballSpeed.y;
135 if ((ballPosition.x >= (GetScreenWidth() - ballRadius)) || (ballPosition.x <= ballRadius)) ballSpeed.x *= -1.0f;
136 if ((ballPosition.y >= (GetScreenHeight() - ballRadius)) || (ballPosition.y <= ballRadius)) ballSpeed.y *= -1.0f;
137 //-----------------------------------------------------
138
139 // Draw
140 //-----------------------------------------------------
141 BeginDrawing();
142
143 if (IsWindowState(FLAG_WINDOW_TRANSPARENT)) ClearBackground(BLANK);
144 else ClearBackground(RAYWHITE);
145
146 DrawCircleV(ballPosition, ballRadius, MAROON);
147 DrawRectangleLinesEx((Rectangle) { 0, 0, (float)GetScreenWidth(), (float)GetScreenHeight() }, 4, RAYWHITE);
148
149 DrawCircleV(GetMousePosition(), 10, DARKBLUE);
150
151 DrawFPS(10, 10);
152
153 DrawText(TextFormat("Screen Size: [%i, %i]", GetScreenWidth(), GetScreenHeight()), 10, 40, 10, GREEN);
154
155 // Draw window state info
156 DrawText("Following flags can be set after window creation:", 10, 60, 10, GRAY);
157 if (IsWindowState(FLAG_FULLSCREEN_MODE)) DrawText("[F] FLAG_FULLSCREEN_MODE: on", 10, 80, 10, LIME);
158 else DrawText("[F] FLAG_FULLSCREEN_MODE: off", 10, 80, 10, MAROON);
159 if (IsWindowState(FLAG_WINDOW_RESIZABLE)) DrawText("[R] FLAG_WINDOW_RESIZABLE: on", 10, 100, 10, LIME);
160 else DrawText("[R] FLAG_WINDOW_RESIZABLE: off", 10, 100, 10, MAROON);
161 if (IsWindowState(FLAG_WINDOW_UNDECORATED)) DrawText("[D] FLAG_WINDOW_UNDECORATED: on", 10, 120, 10, LIME);
162 else DrawText("[D] FLAG_WINDOW_UNDECORATED: off", 10, 120, 10, MAROON);
163 if (IsWindowState(FLAG_WINDOW_HIDDEN)) DrawText("[H] FLAG_WINDOW_HIDDEN: on", 10, 140, 10, LIME);
164 else DrawText("[H] FLAG_WINDOW_HIDDEN: off", 10, 140, 10, MAROON);
165 if (IsWindowState(FLAG_WINDOW_MINIMIZED)) DrawText("[N] FLAG_WINDOW_MINIMIZED: on", 10, 160, 10, LIME);
166 else DrawText("[N] FLAG_WINDOW_MINIMIZED: off", 10, 160, 10, MAROON);
167 if (IsWindowState(FLAG_WINDOW_MAXIMIZED)) DrawText("[M] FLAG_WINDOW_MAXIMIZED: on", 10, 180, 10, LIME);
168 else DrawText("[M] FLAG_WINDOW_MAXIMIZED: off", 10, 180, 10, MAROON);
169 if (IsWindowState(FLAG_WINDOW_UNFOCUSED)) DrawText("[G] FLAG_WINDOW_UNFOCUSED: on", 10, 200, 10, LIME);
170 else DrawText("[U] FLAG_WINDOW_UNFOCUSED: off", 10, 200, 10, MAROON);
171 if (IsWindowState(FLAG_WINDOW_TOPMOST)) DrawText("[T] FLAG_WINDOW_TOPMOST: on", 10, 220, 10, LIME);
172 else DrawText("[T] FLAG_WINDOW_TOPMOST: off", 10, 220, 10, MAROON);
173 if (IsWindowState(FLAG_WINDOW_ALWAYS_RUN)) DrawText("[A] FLAG_WINDOW_ALWAYS_RUN: on", 10, 240, 10, LIME);
174 else DrawText("[A] FLAG_WINDOW_ALWAYS_RUN: off", 10, 240, 10, MAROON);
175 if (IsWindowState(FLAG_VSYNC_HINT)) DrawText("[V] FLAG_VSYNC_HINT: on", 10, 260, 10, LIME);
176 else DrawText("[V] FLAG_VSYNC_HINT: off", 10, 260, 10, MAROON);
177
178 DrawText("Following flags can only be set before window creation:", 10, 300, 10, GRAY);
179 if (IsWindowState(FLAG_WINDOW_HIGHDPI)) DrawText("FLAG_WINDOW_HIGHDPI: on", 10, 320, 10, LIME);
180 else DrawText("FLAG_WINDOW_HIGHDPI: off", 10, 320, 10, MAROON);
181 if (IsWindowState(FLAG_WINDOW_TRANSPARENT)) DrawText("FLAG_WINDOW_TRANSPARENT: on", 10, 340, 10, LIME);
182 else DrawText("FLAG_WINDOW_TRANSPARENT: off", 10, 340, 10, MAROON);
183 if (IsWindowState(FLAG_MSAA_4X_HINT)) DrawText("FLAG_MSAA_4X_HINT: on", 10, 360, 10, LIME);
184 else DrawText("FLAG_MSAA_4X_HINT: off", 10, 360, 10, MAROON);
185
186 EndDrawing();
187 //-----------------------------------------------------
188 }
189
190 // De-Initialization
191 //---------------------------------------------------------
192 CloseWindow(); // Close window and OpenGL context
193 //----------------------------------------------------------
194
195 return 0;
196}
