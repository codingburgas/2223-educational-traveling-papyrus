19#define FLT_MAX 340282346638528859811704183484516925440.0f // Maximum value of a float, from bit pattern 01111111011111111111111111111111

21//------------------------------------------------------------------------------------
22// Program main entry point
23//------------------------------------------------------------------------------------
24int main(void)
25{
26 // Initialization
27 //--------------------------------------------------------------------------------------
28 const int screenWidth = 800;
29 const int screenHeight = 450;
30
31 InitWindow(screenWidth, screenHeight, "raylib [models] example - mesh picking");
32
33 // Define the camera to look into our 3d world
34 Camera camera = { 0 };
35 camera.position = (Vector3){ 20.0f, 20.0f, 20.0f }; // Camera position
36 camera.target = (Vector3){ 0.0f, 8.0f, 0.0f }; // Camera looking at point
37 camera.up = (Vector3){ 0.0f, 1.6f, 0.0f }; // Camera up vector (rotation towards target)
38 camera.fovy = 45.0f; // Camera field-of-view Y
39 camera.projection = CAMERA_PERSPECTIVE; // Camera mode type
40
41 Ray ray = { 0 }; // Picking ray
42
43 Model tower = LoadModel("resources/models/obj/turret.obj"); // Load OBJ model
44 Texture2D texture = LoadTexture("resources/models/obj/turret_diffuse.png"); // Load model texture
45 tower.materials[0].maps[MATERIAL_MAP_DIFFUSE].texture = texture; // Set model diffuse texture
46
47 Vector3 towerPos = { 0.0f, 0.0f, 0.0f }; // Set model position
48 BoundingBox towerBBox = GetMeshBoundingBox(tower.meshes[0]); // Get mesh bounding box
49
50 // Ground quad
51 Vector3 g0 = (Vector3){ -50.0f, 0.0f, -50.0f };
52 Vector3 g1 = (Vector3){ -50.0f, 0.0f, 50.0f };
53 Vector3 g2 = (Vector3){ 50.0f, 0.0f, 50.0f };
54 Vector3 g3 = (Vector3){ 50.0f, 0.0f, -50.0f };
55
56 // Test triangle
57 Vector3 ta = (Vector3){ -25.0f, 0.5f, 0.0f };
58 Vector3 tb = (Vector3){ -4.0f, 2.5f, 1.0f };
59 Vector3 tc = (Vector3){ -8.0f, 6.5f, 0.0f };
60
61 Vector3 bary = { 0.0f, 0.0f, 0.0f };
62
63 // Test sphere
64 Vector3 sp = (Vector3){ -30.0f, 5.0f, 5.0f };
65 float sr = 4.0f;
66
67 SetCameraMode(camera, CAMERA_FREE); // Set a free camera mode
68
69 SetTargetFPS(60); // Set our game to run at 60 frames-per-second
70 //--------------------------------------------------------------------------------------
71 // Main game loop
72 while (!WindowShouldClose()) // Detect window close button or ESC key
73 {
74 // Update
75 //----------------------------------------------------------------------------------
76 UpdateCamera(&camera);
77
78 // Display information about closest hit
79 RayCollision collision = { 0 };
80 char *hitObjectName = "None";
81 collision.distance = FLT_MAX;
82 collision.hit = false;
83 Color cursorColor = WHITE;
84
85 // Get ray and test against objects
86 ray = GetMouseRay(GetMousePosition(), camera);
87
88 // Check ray collision against ground quad
89 RayCollision groundHitInfo = GetRayCollisionQuad(ray, g0, g1, g2, g3);
90
91 if ((groundHitInfo.hit) && (groundHitInfo.distance < collision.distance))
92 {
93 collision = groundHitInfo;
94 cursorColor = GREEN;
95 hitObjectName = "Ground";
96 }
97
98 // Check ray collision against test triangle
99 RayCollision triHitInfo = GetRayCollisionTriangle(ray, ta, tb, tc);
100
101 if ((triHitInfo.hit) && (triHitInfo.distance < collision.distance))
102 {
103 collision = triHitInfo;
104 cursorColor = PURPLE;
105 hitObjectName = "Triangle";
106
107 bary = Vector3Barycenter(collision.point, ta, tb, tc);
108 }
109
110 // Check ray collision against test sphere
111 RayCollision sphereHitInfo = GetRayCollisionSphere(ray, sp, sr);
112
113 if ((sphereHitInfo.hit) && (sphereHitInfo.distance < collision.distance))
114 {
115 collision = sphereHitInfo;
116 cursorColor = ORANGE;
117 hitObjectName = "Sphere";
118 }
119
120 // Check ray collision against bounding box first, before trying the full ray-mesh test
121 RayCollision boxHitInfo = GetRayCollisionBox(ray, towerBBox);
122
123 if ((boxHitInfo.hit) && (boxHitInfo.distance < collision.distance))
124 {
125 collision = boxHitInfo;
126 cursorColor = ORANGE;
127 hitObjectName = "Box";
128
129 // Check ray collision against model meshes
130 RayCollision meshHitInfo = { 0 };
131 for (int m = 0; m < tower.meshCount; m++)
132 {
133 // NOTE: We consider the model.transform for the collision check but
134 // it can be checked against any transform Matrix, used when checking against same
135 // model drawn multiple times with multiple transforms
136 meshHitInfo = GetRayCollisionMesh(ray, tower.meshes[m], tower.transform);
137 if (meshHitInfo.hit)
138 {
139 // Save the closest hit mesh
140 if ((!collision.hit) || (collision.distance > meshHitInfo.distance)) collision = meshHitInfo;
141
142 break; // Stop once one mesh collision is detected, the colliding mesh is m
143 }
144 }
145
146 if (meshHitInfo.hit)
147 {
148 collision = meshHitInfo;
149 cursorColor = ORANGE;
150 hitObjectName = "Mesh";
151 }
152 }
153 //----------------------------------------------------------------------------------
154
155 // Draw
156 //----------------------------------------------------------------------------------
157 BeginDrawing();
158
159 ClearBackground(RAYWHITE);
160
161 BeginMode3D(camera);
162
163 // Draw the tower
164 // WARNING: If scale is different than 1.0f,
165 // not considered by GetRayCollisionModel()
166 DrawModel(tower, towerPos, 1.0f, WHITE);
167
168 // Draw the test triangle
169 DrawLine3D(ta, tb, PURPLE);
170 DrawLine3D(tb, tc, PURPLE);
171 DrawLine3D(tc, ta, PURPLE);
172
173 // Draw the test sphere
174 DrawSphereWires(sp, sr, 8, 8, PURPLE);
175
176 // Draw the mesh bbox if we hit it
177 if (boxHitInfo.hit) DrawBoundingBox(towerBBox, LIME);
178
179 // If we hit something, draw the cursor at the hit point
180 if (collision.hit)
181 {
182 DrawCube(collision.point, 0.3f, 0.3f, 0.3f, cursorColor);
183 DrawCubeWires(collision.point, 0.3f, 0.3f, 0.3f, RED);
184
185 Vector3 normalEnd;
186 normalEnd.x = collision.point.x + collision.normal.x;
187 normalEnd.y = collision.point.y + collision.normal.y;
188 normalEnd.z = collision.point.z + collision.normal.z;
189
190 DrawLine3D(collision.point, normalEnd, RED);
191 }
192
193 DrawRay(ray, MAROON);
194
195 DrawGrid(10, 10.0f);
196
197 EndMode3D();
198
199 // Draw some debug GUI text
200 DrawText(TextFormat("Hit Object: %s", hitObjectName), 10, 50, 10, BLACK);
201
202 if (collision.hit)
203 {
204 int ypos = 70;
205
206 DrawText(TextFormat("Distance: %3.2f", collision.distance), 10, ypos, 10, BLACK);
207
208 DrawText(TextFormat("Hit Pos: %3.2f %3.2f %3.2f",
209 collision.point.x,
210 collision.point.y,
211 collision.point.z), 10, ypos + 15, 10, BLACK);
212
213 DrawText(TextFormat("Hit Norm: %3.2f %3.2f %3.2f",
214 collision.normal.x,
215 collision.normal.y,
216 collision.normal.z), 10, ypos + 30, 10, BLACK);
217
218 if (triHitInfo.hit && TextIsEqual(hitObjectName, "Triangle"))
219 DrawText(TextFormat("Barycenter: %3.2f %3.2f %3.2f", bary.x, bary.y, bary.z), 10, ypos + 45, 10, BLACK);
220 }
221
222 DrawText("Use Mouse to Move Camera", 10, 430, 10, GRAY);
223
224 DrawText("(c) Turret 3D model by Alberto Cano", screenWidth - 200, screenHeight - 20, 10, GRAY);
225
226 DrawFPS(10, 10);
227
228 EndDrawing();
229 //----------------------------------------------------------------------------------
230 }
231
232 // De-Initialization
233 //--------------------------------------------------------------------------------------
234 UnloadModel(tower); // Unload model
235 UnloadTexture(texture); // Unload texture
236
237 CloseWindow(); // Close window and OpenGL context
238 //--------------------------------------------------------------------------------------
239
240 return 0;
241}
