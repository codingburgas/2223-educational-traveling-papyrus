40GLFWbool _glfwPlatformCreateTls(_GLFWtls* tls)
41{
42 assert(tls->posix.allocated == GLFW_FALSE);
43
44 if (pthread_key_create(&tls->posix.key, NULL) != 0)
45 {
46 _glfwInputError(GLFW_PLATFORM_ERROR,
