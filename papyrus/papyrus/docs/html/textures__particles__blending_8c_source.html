20 Vector2 position;

28//------------------------------------------------------------------------------------
29// Program main entry point

31int main(void)
32{
33 // Initialization
34 //--------------------------------------------------------------------------------------
35 const int screenWidth = 800;
36 const int screenHeight = 450;
37
38 InitWindow(screenWidth, screenHeight, "raylib [textures] example - particles blending");
39
40 // Particles pool, reuse them!
41 Particle mouseTail[MAX_PARTICLES] = { 0 };
42
43 // Initialize particles
44 for (int i = 0; i < MAX_PARTICLES; i++)
45 {
46 mouseTail[i].position = (Vector2){ 0, 0 };
47 mouseTail[i].color = (Color){ GetRandomValue(0, 255), GetRandomValue(0, 255), GetRandomValue(0, 255), 255 };
48 mouseTail[i].alpha = 1.0f;
49 mouseTail[i].size = (float)GetRandomValue(1, 30)/20.0f;
50 mouseTail[i].rotation = (float)GetRandomValue(0, 360);
51 mouseTail[i].active = false;
52 }
53
54 float gravity = 3.0f;
55
56 Texture2D smoke = LoadTexture("resources/spark_flame.png");
57
58 int blending = BLEND_ALPHA;
59
60 SetTargetFPS(60);
61 //--------------------------------------------------------------------------------------
62
63 // Main game loop
64 while (!WindowShouldClose()) // Detect window close button or ESC key
65 {
66 // Update
67 //----------------------------------------------------------------------------------
68
69 // Activate one particle every frame and Update active particles
70 // NOTE: Particles initial position should be mouse position when activated
71 // NOTE: Particles fall down with gravity and rotation... and disappear after 2 seconds (alpha = 0)
72 // NOTE: When a particle disappears, active = false and it can be reused.
73 for (int i = 0; i < MAX_PARTICLES; i++)
74 {
75 if (!mouseTail[i].active)
76 {
77 mouseTail[i].active = true;
78 mouseTail[i].alpha = 1.0f;
79 mouseTail[i].position = GetMousePosition();
80 i = MAX_PARTICLES;
81 }
82 }
83
84 for (int i = 0; i < MAX_PARTICLES; i++)
85 {
86 if (mouseTail[i].active)
87 {
88 mouseTail[i].position.y += gravity/2;
89 mouseTail[i].alpha -= 0.005f;
90
91 if (mouseTail[i].alpha <= 0.0f) mouseTail[i].active = false;
92
93 mouseTail[i].rotation += 2.0f;
94 }
95 }
96
97 if (IsKeyPressed(KEY_SPACE))
98 {
99 if (blending == BLEND_ALPHA) blending = BLEND_ADDITIVE;
100 else blending = BLEND_ALPHA;
101 }
102 //----------------------------------------------------------------------------------
103
104 // Draw
105 //----------------------------------------------------------------------------------
106 BeginDrawing();
107
108 ClearBackground(DARKGRAY);
109
110 BeginBlendMode(blending);
111
112 // Draw active particles
113 for (int i = 0; i < MAX_PARTICLES; i++)
114 {
115 if (mouseTail[i].active) DrawTexturePro(smoke, (Rectangle){ 0.0f, 0.0f, (float)smoke.width, (float)smoke.height },
116 (Rectangle){ mouseTail[i].position.x, mouseTail[i].position.y, smoke.width*mouseTail[i].size, smoke.height*mouseTail[i].size },
117 (Vector2){ (float)(smoke.width*mouseTail[i].size/2.0f), (float)(smoke.height*mouseTail[i].size/2.0f) }, mouseTail[i].rotation,
118 Fade(mouseTail[i].color, mouseTail[i].alpha));
119 }
120
121 EndBlendMode();
122
123 DrawText("PRESS SPACE to CHANGE BLENDING MODE", 180, 20, 20, BLACK);
124
125 if (blending == BLEND_ALPHA) DrawText("ALPHA BLENDING", 290, screenHeight - 40, 20, BLACK);
126 else DrawText("ADDITIVE BLENDING", 280, screenHeight - 40, 20, RAYWHITE);
127
128 EndDrawing();
129 //----------------------------------------------------------------------------------
130 }
131
132 // De-Initialization
133 //--------------------------------------------------------------------------------------
134 UnloadTexture(smoke);
135
136 CloseWindow(); // Close window and OpenGL context
137 //--------------------------------------------------------------------------------------
138
139 return 0;
140}
