21// Scene drawing
22void DrawScene(void)
23{
24 int count = 5;
25 float spacing = 4;
26
27 // Grid of cube trees on a plane to make a "world"
28 DrawPlane((Vector3){ 0, 0, 0 }, (Vector2){ 50, 50 }, BEIGE); // Simple world plane
29
30 for (float x = -count*spacing; x <= count*spacing; x += spacing)
31 {
32 for (float z = -count*spacing; z <= count*spacing; z += spacing)
33 {
34 DrawCube((Vector3) { x, 1.5f, z }, 1, 1, 1, LIME);
35 DrawCube((Vector3) { x, 0.5f, z }, 0.25f, 1, 0.25f, BROWN);
36 }
37 }
38
39 // Draw a cube at each player's position
40 DrawCube(cameraPlayer1.position, 1, 1, 1, RED);
41 DrawCube(cameraPlayer2.position, 1, 1, 1, BLUE);
42}

72 RenderTexture screenPlayer2 = LoadRenderTexture(screenWidth / 2, screenHeight);
73
74 // Build a flipped rectangle the size of the split view to use for drawing later
75 Rectangle splitScreenRect = { 0.0f, 0.0f, (float)screenPlayer1.texture.width, (float)-screenPlayer1.texture.height };
76
77 SetTargetFPS(60); // Set our game to run at 60 frames-per-second
78 //--------------------------------------------------------------------------------------
79
80 // Main game loop
81 while (!WindowShouldClose()) // Detect window close button or ESC key
82 {
83 // Update
84 //----------------------------------------------------------------------------------
85 // If anyone moves this frame, how far will they move based on the time since the last frame
86 // this moves thigns at 10 world units per second, regardless of the actual FPS
87 float offsetThisFrame = 10.0f*GetFrameTime();
88
89 // Move Player1 forward and backwards (no turning)
90 if (IsKeyDown(KEY_W))
91 {
92 cameraPlayer1.position.z += offsetThisFrame;
93 cameraPlayer1.target.z += offsetThisFrame;
94 }
95 else if (IsKeyDown(KEY_S))
96 {
97 cameraPlayer1.position.z -= offsetThisFrame;
98 cameraPlayer1.target.z -= offsetThisFrame;
99 }
100
101 // Move Player2 forward and backwards (no turning)
102 if (IsKeyDown(KEY_UP))
103 {
104 cameraPlayer2.position.x += offsetThisFrame;
105 cameraPlayer2.target.x += offsetThisFrame;
106 }
107 else if (IsKeyDown(KEY_DOWN))
108 {
109 cameraPlayer2.position.x -= offsetThisFrame;
110 cameraPlayer2.target.x -= offsetThisFrame;
111 }
112 //----------------------------------------------------------------------------------
113
114 // Draw
115 //----------------------------------------------------------------------------------
116 // Draw Player1 view to the render texture
117 BeginTextureMode(screenPlayer1);
118 ClearBackground(SKYBLUE);
119 BeginMode3D(cameraPlayer1);
120 DrawScene();
121 EndMode3D();
122 DrawText("PLAYER1 W/S to move", 10, 10, 20, RED);
123 EndTextureMode();
124
125 // Draw Player2 view to the render texture
126 BeginTextureMode(screenPlayer2);
127 ClearBackground(SKYBLUE);
128 BeginMode3D(cameraPlayer2);
129 DrawScene();
130 EndMode3D();
131 DrawText("PLAYER2 UP/DOWN to move", 10, 10, 20, BLUE);
132 EndTextureMode();
133
134 // Draw both views render textures to the screen side by side
135 BeginDrawing();
136 ClearBackground(BLACK);
137 DrawTextureRec(screenPlayer1.texture, splitScreenRect, (Vector2){ 0, 0 }, WHITE);
138 DrawTextureRec(screenPlayer2.texture, splitScreenRect, (Vector2){ screenWidth/2.0f, 0 }, WHITE);
139 EndDrawing();
140 }
141
142 // De-Initialization
143 //--------------------------------------------------------------------------------------
144 UnloadRenderTexture(screenPlayer1); // Unload render texture
145 UnloadRenderTexture(screenPlayer2); // Unload render texture
146
147 CloseWindow(); // Close window and OpenGL context
148 //--------------------------------------------------------------------------------------
149
150 return 0;
151}
