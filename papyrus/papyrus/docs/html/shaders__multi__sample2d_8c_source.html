3* raylib [shaders] example - Multiple sample2D with default batch system
4*
5* NOTE: This example requires raylib OpenGL 3.3 or ES2 versions for shaders support,
6* OpenGL 1.1 does not support shaders, recompile raylib to OpenGL 3.3 version.
7*
8* NOTE: Shaders used in this example are #version 330 (OpenGL 3.3), to test this example
9* on OpenGL ES 2.0 platforms (Android, Raspberry Pi, HTML5), use #version 100 shaders

49 Shader shader = LoadShader(0, TextFormat("resources/shaders/glsl%i/color_mix.fs", GLSL_VERSION));
50
51 // Get an additional sampler2D location to be enabled on drawing
52 int texBlueLoc = GetShaderLocation(shader, "texture1");
53
54 // Get shader uniform for divider
55 int dividerLoc = GetShaderLocation(shader, "divider");
56 float dividerValue = 0.5f;
57
58 SetTargetFPS(60); // Set our game to run at 60 frames-per-second
59 //--------------------------------------------------------------------------------------
60
61 // Main game loop
62 while (!WindowShouldClose()) // Detect window close button or ESC key
63 {
64 // Update
65 //----------------------------------------------------------------------------------
66 if (IsKeyDown(KEY_RIGHT)) dividerValue += 0.01f;
67 else if (IsKeyDown(KEY_LEFT)) dividerValue -= 0.01f;
68
69 if (dividerValue < 0.0f) dividerValue = 0.0f;
70 else if (dividerValue > 1.0f) dividerValue = 1.0f;
71
72 SetShaderValue(shader, dividerLoc, &dividerValue, SHADER_UNIFORM_FLOAT);
73 //----------------------------------------------------------------------------------
74
75 // Draw
76 //----------------------------------------------------------------------------------
77 BeginDrawing();
78
79 ClearBackground(RAYWHITE);
80
81 BeginShaderMode(shader);
82
83 // WARNING: Additional samplers are enabled for all draw calls in the batch,
84 // EndShaderMode() forces batch drawing and consequently resets active textures
85 // to let other sampler2D to be activated on consequent drawings (if required)
86 SetShaderValueTexture(shader, texBlueLoc, texBlue);
87
88 // We are drawing texRed using default sampler2D texture0 but
89 // an additional texture units is enabled for texBlue (sampler2D texture1)
90 DrawTexture(texRed, 0, 0, WHITE);
91
92 EndShaderMode();
93
94 DrawText("Use KEY_LEFT/KEY_RIGHT to move texture mixing in shader!", 80, GetScreenHeight() - 40, 20, RAYWHITE);
95
96 EndDrawing();
97 //----------------------------------------------------------------------------------
98 }
99
100 // De-Initialization
101 //--------------------------------------------------------------------------------------
102 UnloadShader(shader); // Unload shader
103 UnloadTexture(texRed); // Unload texture
104 UnloadTexture(texBlue); // Unload texture
105
106 CloseWindow(); // Close window and OpenGL context
107 //--------------------------------------------------------------------------------------
108
109 return 0;
110}
