40 camera.target = (Vector3){ 0.0f, 0.0f, 0.7f }; // Camera looking at point
41 camera.up = (Vector3){ 0.0f, 1.0f, 0.0f }; // Camera up vector (rotation towards target)
42 camera.fovy = 65.0f; // Camera field-of-view Y
43
44 SetCameraMode(camera, CAMERA_FREE); // Set camera mode
45
46 // Load raymarching shader
47 // NOTE: Defining 0 (NULL) for vertex shader forces usage of internal default vertex shader
48 Shader shader = LoadShader(0, TextFormat("resources/shaders/glsl%i/raymarching.fs", GLSL_VERSION));
49
50 // Get shader locations for required uniforms
51 int viewEyeLoc = GetShaderLocation(shader, "viewEye");
52 int viewCenterLoc = GetShaderLocation(shader, "viewCenter");
53 int runTimeLoc = GetShaderLocation(shader, "runTime");
54 int resolutionLoc = GetShaderLocation(shader, "resolution");
55
56 float resolution[2] = { (float)screenWidth, (float)screenHeight };
57 SetShaderValue(shader, resolutionLoc, resolution, SHADER_UNIFORM_VEC2);
58
59 float runTime = 0.0f;
60
61 SetTargetFPS(60); // Set our game to run at 60 frames-per-second
62 //--------------------------------------------------------------------------------------
63
64 // Main game loop
65 while (!WindowShouldClose()) // Detect window close button or ESC key
66 {
67 // Update
68 //----------------------------------------------------------------------------------
69 UpdateCamera(&camera);
70
71 float cameraPos[3] = { camera.position.x, camera.position.y, camera.position.z };
72 float cameraTarget[3] = { camera.target.x, camera.target.y, camera.target.z };
73
74 float deltaTime = GetFrameTime();
75 runTime += deltaTime;
76
77 // Set shader required uniform values
78 SetShaderValue(shader, viewEyeLoc, cameraPos, SHADER_UNIFORM_VEC3);
79 SetShaderValue(shader, viewCenterLoc, cameraTarget, SHADER_UNIFORM_VEC3);
80 SetShaderValue(shader, runTimeLoc, &runTime, SHADER_UNIFORM_FLOAT);
81
82 // Check if screen is resized
83 if (IsWindowResized())
84 {
85 float resolution[2] = { (float)GetScreenWidth(), (float)GetScreenHeight() };
86 SetShaderValue(shader, resolutionLoc, resolution, SHADER_UNIFORM_VEC2);
87 }
88 //----------------------------------------------------------------------------------
89
90 // Draw
91 //----------------------------------------------------------------------------------
92 BeginDrawing();
93
94 ClearBackground(RAYWHITE);
95
96 // We only draw a white full-screen rectangle,
97 // frame is generated in shader using raymarching
98 BeginShaderMode(shader);
99 DrawRectangle(0, 0, GetScreenWidth(), GetScreenHeight(), WHITE);
100 EndShaderMode();
101
102 DrawText("(c) Raymarching shader by IÃ±igo Quilez. MIT License.", GetScreenWidth() - 280, GetScreenHeight() - 20, 10, BLACK);
103
104 EndDrawing();
105 //----------------------------------------------------------------------------------
106 }
107
108 // De-Initialization
109 //--------------------------------------------------------------------------------------

111
112 CloseWindow(); // Close window and OpenGL context
113 //--------------------------------------------------------------------------------------
114
115 return 0;
