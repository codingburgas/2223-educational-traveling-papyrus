16#define MAX_CIRCLES 64

18typedef struct {
19 Vector2 position;
20 float radius;
21 float alpha;
22 float speed;
23 Color color;
24} CircleWave;

26//------------------------------------------------------------------------------------
27// Program main entry point
28//------------------------------------------------------------------------------------
29int main(void)
30{
31 // Initialization
32 //--------------------------------------------------------------------------------------
33 const int screenWidth = 800;
34 const int screenHeight = 450;
35
36 SetConfigFlags(FLAG_MSAA_4X_HINT); // NOTE: Try to enable MSAA 4X
37
38 InitWindow(screenWidth, screenHeight, "raylib [audio] example - module playing (streaming)");
39
40 InitAudioDevice(); // Initialize audio device
41
42 Color colors[14] = { ORANGE, RED, GOLD, LIME, BLUE, VIOLET, BROWN, LIGHTGRAY, PINK,
43 YELLOW, GREEN, SKYBLUE, PURPLE, BEIGE };
44
45 // Creates some circles for visual effect
46 CircleWave circles[MAX_CIRCLES] = { 0 };
47
48 for (int i = MAX_CIRCLES - 1; i >= 0; i--)
49 {
50 circles[i].alpha = 0.0f;
51 circles[i].radius = (float)GetRandomValue(10, 40);
52 circles[i].position.x = (float)GetRandomValue((int)circles[i].radius, (int)(screenWidth - circles[i].radius));
53 circles[i].position.y = (float)GetRandomValue((int)circles[i].radius, (int)(screenHeight - circles[i].radius));
54 circles[i].speed = (float)GetRandomValue(1, 100)/2000.0f;
55 circles[i].color = colors[GetRandomValue(0, 13)];
56 }
57
58 Music music = LoadMusicStream("resources/mini1111.xm");
59 music.looping = false;
60 float pitch = 1.0f;
61
62 PlayMusicStream(music);
63
64 float timePlayed = 0.0f;
65 bool pause = false;
66
67 SetTargetFPS(60); // Set our game to run at 60 frames-per-second
68 //--------------------------------------------------------------------------------------
69
70 // Main game loop
71 while (!WindowShouldClose()) // Detect window close button or ESC key
72 {
73 // Update
74 //----------------------------------------------------------------------------------
75 UpdateMusicStream(music); // Update music buffer with new stream data
76
77 // Restart music playing (stop and play)
78 if (IsKeyPressed(KEY_SPACE))
79 {
80 StopMusicStream(music);
81 PlayMusicStream(music);
82 }
83
84 // Pause/Resume music playing
85 if (IsKeyPressed(KEY_P))
86 {
87 pause = !pause;
88
89 if (pause) PauseMusicStream(music);
90 else ResumeMusicStream(music);
91 }
92
93 if (IsKeyDown(KEY_DOWN)) pitch -= 0.01f;
94 else if (IsKeyDown(KEY_UP)) pitch += 0.01f;
95
96 SetMusicPitch(music, pitch);
97
98 // Get timePlayed scaled to bar dimensions
99 timePlayed = GetMusicTimePlayed(music)/GetMusicTimeLength(music)*(screenWidth - 40);
100
101 // Color circles animation
102 for (int i = MAX_CIRCLES - 1; (i >= 0) && !pause; i--)
103 {
104 circles[i].alpha += circles[i].speed;
105 circles[i].radius += circles[i].speed*10.0f;
106
107 if (circles[i].alpha > 1.0f) circles[i].speed *= -1;
108
109 if (circles[i].alpha <= 0.0f)
110 {
111 circles[i].alpha = 0.0f;
112 circles[i].radius = (float)GetRandomValue(10, 40);
113 circles[i].position.x = (float)GetRandomValue((int)circles[i].radius, (int)(screenWidth - circles[i].radius));
114 circles[i].position.y = (float)GetRandomValue((int)circles[i].radius, (int)(screenHeight - circles[i].radius));
115 circles[i].color = colors[GetRandomValue(0, 13)];
116 circles[i].speed = (float)GetRandomValue(1, 100)/2000.0f;
117 }
118 }
119 //----------------------------------------------------------------------------------
120
121 // Draw
122 //----------------------------------------------------------------------------------
123 BeginDrawing();
124
125 ClearBackground(RAYWHITE);
126
127 for (int i = MAX_CIRCLES - 1; i >= 0; i--)
128 {
129 DrawCircleV(circles[i].position, circles[i].radius, Fade(circles[i].color, circles[i].alpha));
130 }
131
132 // Draw time bar
133 DrawRectangle(20, screenHeight - 20 - 12, screenWidth - 40, 12, LIGHTGRAY);
134 DrawRectangle(20, screenHeight - 20 - 12, (int)timePlayed, 12, MAROON);
135 DrawRectangleLines(20, screenHeight - 20 - 12, screenWidth - 40, 12, GRAY);
136
137 EndDrawing();
138 //----------------------------------------------------------------------------------
139 }
140
141 // De-Initialization
142 //--------------------------------------------------------------------------------------
143 UnloadMusicStream(music); // Unload music stream buffers from RAM
144
145 CloseAudioDevice(); // Close audio device (music streaming is automatically stopped)
146
147 CloseWindow(); // Close window and OpenGL context
148 //--------------------------------------------------------------------------------------
149
150 return 0;
151}
