31int main(void)
32{
33 // Initialization
34 //--------------------------------------------------------------------------------------
35 const int screenWidth = 800;
36 const int screenHeight = 450;
37
38 InitWindow(screenWidth, screenHeight, "raylib [models] example - skybox loading and drawing");
39
40 // Define the camera to look into our 3d world
41 Camera camera = { { 1.0f, 1.0f, 1.0f }, { 4.0f, 1.0f, 4.0f }, { 0.0f, 1.0f, 0.0f }, 45.0f, 0 };
42
43 // Load skybox model
44 Mesh cube = GenMeshCube(1.0f, 1.0f, 1.0f);
45 Model skybox = LoadModelFromMesh(cube);
46
47 bool useHDR = true;
48
49 // Load skybox shader and set required locations
50 // NOTE: Some locations are automatically set at shader loading
51 skybox.materials[0].shader = LoadShader(TextFormat("resources/shaders/glsl%i/skybox.vs", GLSL_VERSION),
52 TextFormat("resources/shaders/glsl%i/skybox.fs", GLSL_VERSION));
53
54 SetShaderValue(skybox.materials[0].shader, GetShaderLocation(skybox.materials[0].shader, "environmentMap"), (int[1]){ MATERIAL_MAP_CUBEMAP }, SHADER_UNIFORM_INT);
55 SetShaderValue(skybox.materials[0].shader, GetShaderLocation(skybox.materials[0].shader, "doGamma"), (int[1]) { useHDR ? 1 : 0 }, SHADER_UNIFORM_INT);
56 SetShaderValue(skybox.materials[0].shader, GetShaderLocation(skybox.materials[0].shader, "vflipped"), (int[1]){ useHDR ? 1 : 0 }, SHADER_UNIFORM_INT);
57
58 // Load cubemap shader and setup required shader locations
59 Shader shdrCubemap = LoadShader(TextFormat("resources/shaders/glsl%i/cubemap.vs", GLSL_VERSION),
60 TextFormat("resources/shaders/glsl%i/cubemap.fs", GLSL_VERSION));
61
62 SetShaderValue(shdrCubemap, GetShaderLocation(shdrCubemap, "equirectangularMap"), (int[1]){ 0 }, SHADER_UNIFORM_INT);
63
64 char skyboxFileName[256] = { 0 };
65
66 Texture2D panorama;
67
68 if (useHDR)
69 {
70 TextCopy(skyboxFileName, "resources/dresden_square_2k.hdr");
71
72 // Load HDR panorama (sphere) texture
73 panorama = LoadTexture(skyboxFileName);
74
75 // Generate cubemap (texture with 6 quads-cube-mapping) from panorama HDR texture
76 // NOTE 1: New texture is generated rendering to texture, shader calculates the sphere->cube coordinates mapping
77 // NOTE 2: It seems on some Android devices WebGL, fbo does not properly support a FLOAT-based attachment,
78 // despite texture can be successfully created.. so using PIXELFORMAT_UNCOMPRESSED_R8G8B8A8 instead of PIXELFORMAT_UNCOMPRESSED_R32G32B32A32
79 skybox.materials[0].maps[MATERIAL_MAP_CUBEMAP].texture = GenTextureCubemap(shdrCubemap, panorama, 1024, PIXELFORMAT_UNCOMPRESSED_R8G8B8A8);
80
81 //UnloadTexture(panorama); // Texture not required anymore, cubemap already generated
82 }
83 else
84 {
85 Image img = LoadImage("resources/skybox.png");
86 skybox.materials[0].maps[MATERIAL_MAP_CUBEMAP].texture = LoadTextureCubemap(img, CUBEMAP_LAYOUT_AUTO_DETECT); // CUBEMAP_LAYOUT_PANORAMA
87 UnloadImage(img);
88 }
89
90 SetCameraMode(camera, CAMERA_FIRST_PERSON); // Set a first person camera mode
91
92 SetTargetFPS(60); // Set our game to run at 60 frames-per-second
93 //--------------------------------------------------------------------------------------
94
95 // Main game loop
96 while (!WindowShouldClose()) // Detect window close button or ESC key
97 {
98 // Update
99 //----------------------------------------------------------------------------------
100 UpdateCamera(&camera);
101
102 // Load new cubemap texture on drag&drop
103 if (IsFileDropped())
104 {
105 FilePathList droppedFiles = LoadDroppedFiles();
106
107 if (droppedFiles.count == 1) // Only support one file dropped
108 {
109 if (IsFileExtension(droppedFiles.paths[0], ".png;.jpg;.hdr;.bmp;.tga"))
110 {
111 // Unload current cubemap texture and load new one
112 UnloadTexture(skybox.materials[0].maps[MATERIAL_MAP_CUBEMAP].texture);
113 if (useHDR)
114 {
115 Texture2D panorama = LoadTexture(droppedFiles.paths[0]);
116
117 // Generate cubemap from panorama texture
118 skybox.materials[0].maps[MATERIAL_MAP_CUBEMAP].texture = GenTextureCubemap(shdrCubemap, panorama, 1024, PIXELFORMAT_UNCOMPRESSED_R8G8B8A8);
119 UnloadTexture(panorama);
120 }
121 else
122 {
123 Image img = LoadImage(droppedFiles.paths[0]);
124 skybox.materials[0].maps[MATERIAL_MAP_CUBEMAP].texture = LoadTextureCubemap(img, CUBEMAP_LAYOUT_AUTO_DETECT);
125 UnloadImage(img);
126 }
127
128 TextCopy(skyboxFileName, droppedFiles.paths[0]);
129 }
130 }
131
132 UnloadDroppedFiles(droppedFiles); // Unload filepaths from memory
133 }
134 //----------------------------------------------------------------------------------
135
136 // Draw
137 //----------------------------------------------------------------------------------
138 BeginDrawing();
139
140 ClearBackground(RAYWHITE);
141
142 BeginMode3D(camera);
143
144 // We are inside the cube, we need to disable backface culling!
145 rlDisableBackfaceCulling();
146 rlDisableDepthMask();
147 DrawModel(skybox, (Vector3){0, 0, 0}, 1.0f, WHITE);
148 rlEnableBackfaceCulling();
149 rlEnableDepthMask();
150
151 DrawGrid(10, 1.0f);
152
153 EndMode3D();
154
155 //DrawTextureEx(panorama, (Vector2){ 0, 0 }, 0.0f, 0.5f, WHITE);
156
157 if (useHDR) DrawText(TextFormat("Panorama image from hdrihaven.com: %s", GetFileName(skyboxFileName)), 10, GetScreenHeight() - 20, 10, BLACK);
158 else DrawText(TextFormat(": %s", GetFileName(skyboxFileName)), 10, GetScreenHeight() - 20, 10, BLACK);
159
160 DrawFPS(10, 10);
161
162 EndDrawing();
163 //----------------------------------------------------------------------------------
164 }
165
166 // De-Initialization
167 //--------------------------------------------------------------------------------------
168 UnloadShader(skybox.materials[0].shader);
169 UnloadTexture(skybox.materials[0].maps[MATERIAL_MAP_CUBEMAP].texture);
170
171 UnloadModel(skybox); // Unload skybox model
172
173 CloseWindow(); // Close window and OpenGL context
174 //--------------------------------------------------------------------------------------
175
176 return 0;
177}

179// Generate cubemap texture from HDR texture
180static TextureCubemap GenTextureCubemap(Shader shader, Texture2D panorama, int size, int format)
181{
182 TextureCubemap cubemap = { 0 };
183
184 rlDisableBackfaceCulling(); // Disable backface culling to render inside the cube
185
186 // STEP 1: Setup framebuffer
187 //------------------------------------------------------------------------------------------
188 unsigned int rbo = rlLoadTextureDepth(size, size, true);
189 cubemap.id = rlLoadTextureCubemap(0, size, format);
190
191 unsigned int fbo = rlLoadFramebuffer(size, size);
192 rlFramebufferAttach(fbo, rbo, RL_ATTACHMENT_DEPTH, RL_ATTACHMENT_RENDERBUFFER, 0);
193 rlFramebufferAttach(fbo, cubemap.id, RL_ATTACHMENT_COLOR_CHANNEL0, RL_ATTACHMENT_CUBEMAP_POSITIVE_X, 0);
194
195 // Check if framebuffer is complete with attachments (valid)
196 if (rlFramebufferComplete(fbo)) TraceLog(LOG_INFO, "FBO: [ID %i] Framebuffer object created successfully", fbo);
197 //------------------------------------------------------------------------------------------
198
199 // STEP 2: Draw to framebuffer
200 //------------------------------------------------------------------------------------------
201 // NOTE: Shader is used to convert HDR equirectangular environment map to cubemap equivalent (6 faces)
202 rlEnableShader(shader.id);
203
204 // Define projection matrix and send it to shader
205 Matrix matFboProjection = MatrixPerspective(90.0*DEG2RAD, 1.0, RL_CULL_DISTANCE_NEAR, RL_CULL_DISTANCE_FAR);
206 rlSetUniformMatrix(shader.locs[SHADER_LOC_MATRIX_PROJECTION], matFboProjection);
207
208 // Define view matrix for every side of the cubemap
209 Matrix fboViews[6] = {
210 MatrixLookAt((Vector3){ 0.0f, 0.0f, 0.0f }, (Vector3){ 1.0f, 0.0f, 0.0f }, (Vector3){ 0.0f, -1.0f, 0.0f }),
211 MatrixLookAt((Vector3){ 0.0f, 0.0f, 0.0f }, (Vector3){ -1.0f, 0.0f, 0.0f }, (Vector3){ 0.0f, -1.0f, 0.0f }),
212 MatrixLookAt((Vector3){ 0.0f, 0.0f, 0.0f }, (Vector3){ 0.0f, 1.0f, 0.0f }, (Vector3){ 0.0f, 0.0f, 1.0f }),
213 MatrixLookAt((Vector3){ 0.0f, 0.0f, 0.0f }, (Vector3){ 0.0f, -1.0f, 0.0f }, (Vector3){ 0.0f, 0.0f, -1.0f }),
214 MatrixLookAt((Vector3){ 0.0f, 0.0f, 0.0f }, (Vector3){ 0.0f, 0.0f, 1.0f }, (Vector3){ 0.0f, -1.0f, 0.0f }),
215 MatrixLookAt((Vector3){ 0.0f, 0.0f, 0.0f }, (Vector3){ 0.0f, 0.0f, -1.0f }, (Vector3){ 0.0f, -1.0f, 0.0f })
216 };
217
218 rlViewport(0, 0, size, size); // Set viewport to current fbo dimensions
219
220 // Activate and enable texture for drawing to cubemap faces
221 rlActiveTextureSlot(0);
222 rlEnableTexture(panorama.id);
223
224 for (int i = 0; i < 6; i++)
225 {
226 // Set the view matrix for the current cube face
227 rlSetUniformMatrix(shader.locs[SHADER_LOC_MATRIX_VIEW], fboViews[i]);
228
229 // Select the current cubemap face attachment for the fbo
230 // WARNING: This function by default enables->attach->disables fbo!!!
231 rlFramebufferAttach(fbo, cubemap.id, RL_ATTACHMENT_COLOR_CHANNEL0, RL_ATTACHMENT_CUBEMAP_POSITIVE_X + i, 0);
232 rlEnableFramebuffer(fbo);
233
234 // Load and draw a cube, it uses the current enabled texture
235 rlClearScreenBuffers();
236 rlLoadDrawCube();
237
238 // ALTERNATIVE: Try to use internal batch system to draw the cube instead of rlLoadDrawCube
239 // for some reason this method does not work, maybe due to cube triangles definition? normals pointing out?
240 // TODO: Investigate this issue...
241 //rlSetTexture(panorama.id); // WARNING: It must be called after enabling current framebuffer if using internal batch system!
242 //rlClearScreenBuffers();
243 //DrawCubeV(Vector3Zero(), Vector3One(), WHITE);
244 //rlDrawRenderBatchActive();
245 }
246 //------------------------------------------------------------------------------------------
247
248 // STEP 3: Unload framebuffer and reset state
249 //------------------------------------------------------------------------------------------
250 rlDisableShader(); // Unbind shader
251 rlDisableTexture(); // Unbind texture
252 rlDisableFramebuffer(); // Unbind framebuffer
253 rlUnloadFramebuffer(fbo); // Unload framebuffer (and automatically attached depth texture/renderbuffer)
254
255 // Reset viewport dimensions to default
256 rlViewport(0, 0, rlGetFramebufferWidth(), rlGetFramebufferHeight());
257 rlEnableBackfaceCulling();
258 //------------------------------------------------------------------------------------------
259
260 cubemap.width = size;
261 cubemap.height = size;
262 cubemap.mipmaps = 1;
263 cubemap.format = format;
264
265 return cubemap;
266}
