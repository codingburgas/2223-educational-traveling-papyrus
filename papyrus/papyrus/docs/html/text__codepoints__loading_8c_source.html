127// Remove codepoint duplicates if requested
128// WARNING: This process could be a bit slow if there text to process is very long
129static int *CodepointRemoveDuplicates(int *codepoints, int codepointCount, int *codepointsResultCount)
130{
131 int codepointsNoDupsCount = codepointCount;
132 int *codepointsNoDups = (int *)calloc(codepointCount, sizeof(int));
133 memcpy(codepointsNoDups, codepoints, codepointCount*sizeof(int));
134
135 // Remove duplicates
136 for (int i = 0; i < codepointsNoDupsCount; i++)
137 {
138 for (int j = i + 1; j < codepointsNoDupsCount; j++)
139 {
140 if (codepointsNoDups[i] == codepointsNoDups[j])
141 {
142 for (int k = j; k < codepointsNoDupsCount; k++) codepointsNoDups[k] = codepointsNoDups[k + 1];
143
144 codepointsNoDupsCount--;
145 j--;
146 }
147 }
148 }
149
150 // NOTE: The size of codepointsNoDups is the same as original array but
151 // only required positions are filled (codepointsNoDupsCount)
152
153 *codepointsResultCount = codepointsNoDupsCount;
154 return codepointsNoDups;
155}
