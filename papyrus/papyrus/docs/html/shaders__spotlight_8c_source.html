3* raylib [shaders] example - Simple shader mask
4*
5* Example originally created with raylib 2.5, last time updated with raylib 3.7

7* Example contributed by Chris Camacho (@chriscamacho) and reviewed by Ramon Santamaria (@raysan5)

12* Copyright (c) 2019-2022 Chris Camacho (@chriscamacho) and Ramon Santamaria (@raysan5)
13*
14********************************************************************************************
15*
16* The shader makes alpha holes in the forground to give the appearance of a top
17* down look at a spotlight casting a pool of light...
18*
19* The right hand side of the screen there is just enough light to see whats
20* going on without the spot light, great for a stealth type game where you
21* have to avoid the spotlights.
22*
23* The left hand side of the screen is in pitch dark except for where the spotlights are.
24*
25* Although this example doesn't scale like the letterbox example, you could integrate
26* the two techniques, but by scaling the actual colour of the render texture rather
27* than using alpha as a mask.
28*
29********************************************************************************************/

68//------------------------------------------------------------------------------------
69// Program main entry point
70//------------------------------------------------------------------------------------
71int main(void)
72{
73 // Initialization
74 //--------------------------------------------------------------------------------------
75 const int screenWidth = 800;
76 const int screenHeight = 450;
77
78 InitWindow(screenWidth, screenHeight, "raylib - shader spotlight");
79 HideCursor();
80
81 Texture texRay = LoadTexture("resources/raysan.png");
82
83 Star stars[MAX_STARS] = { 0 };
84
85 for (int n = 0; n < MAX_STARS; n++) ResetStar(&stars[n]);
86
87 // Progress all the stars on, so they don't all start in the centre
88 for (int m = 0; m < screenWidth/2.0; m++)
89 {
90 for (int n = 0; n < MAX_STARS; n++) UpdateStar(&stars[n]);
91 }
92
93 int frameCounter = 0;
94
95 // Use default vert shader
96 Shader shdrSpot = LoadShader(0, TextFormat("resources/shaders/glsl%i/spotlight.fs", GLSL_VERSION));
97
98 // Get the locations of spots in the shader
99 Spot spots[MAX_SPOTS];
100
101 for (int i = 0; i < MAX_SPOTS; i++)
102 {
103 char posName[32] = "spots[x].pos\0";
104 char innerName[32] = "spots[x].inner\0";
105 char radiusName[32] = "spots[x].radius\0";
106
107 posName[6] = '0' + i;
108 innerName[6] = '0' + i;
109 radiusName[6] = '0' + i;
110
111 spots[i].posLoc = GetShaderLocation(shdrSpot, posName);
112 spots[i].innerLoc = GetShaderLocation(shdrSpot, innerName);
113 spots[i].radiusLoc = GetShaderLocation(shdrSpot, radiusName);
114
115 }
116
117 // Tell the shader how wide the screen is so we can have
118 // a pitch black half and a dimly lit half.
119 unsigned int wLoc = GetShaderLocation(shdrSpot, "screenWidth");
120 float sw = (float)GetScreenWidth();
121 SetShaderValue(shdrSpot, wLoc, &sw, SHADER_UNIFORM_FLOAT);
122
123 // Randomize the locations and velocities of the spotlights
124 // and initialize the shader locations
125 for (int i = 0; i < MAX_SPOTS; i++)
126 {
127 spots[i].pos.x = (float)GetRandomValue(64, screenWidth - 64);
128 spots[i].pos.y = (float)GetRandomValue(64, screenHeight - 64);
129 spots[i].vel = (Vector2){ 0, 0 };
130
131 while ((fabs(spots[i].vel.x) + fabs(spots[i].vel.y)) < 2)
132 {
133 spots[i].vel.x = GetRandomValue(-400, 40) / 10.0f;
134 spots[i].vel.y = GetRandomValue(-400, 40) / 10.0f;
135 }
136
137 spots[i].inner = 28.0f * (i + 1);
138 spots[i].radius = 48.0f * (i + 1);
139
140 SetShaderValue(shdrSpot, spots[i].posLoc, &spots[i].pos.x, SHADER_UNIFORM_VEC2);
141 SetShaderValue(shdrSpot, spots[i].innerLoc, &spots[i].inner, SHADER_UNIFORM_FLOAT);
142 SetShaderValue(shdrSpot, spots[i].radiusLoc, &spots[i].radius, SHADER_UNIFORM_FLOAT);
143 }
144
145 SetTargetFPS(60); // Set to run at 60 frames-per-second
146 //--------------------------------------------------------------------------------------
147
148 // Main game loop
149 while (!WindowShouldClose()) // Detect window close button or ESC key
150 {
151 // Update
152 //----------------------------------------------------------------------------------
153 frameCounter++;
154
155 // Move the stars, resetting them if the go offscreen
156 for (int n = 0; n < MAX_STARS; n++) UpdateStar(&stars[n]);
157
158 // Update the spots, send them to the shader
159 for (int i = 0; i < MAX_SPOTS; i++)
160 {
161 if (i == 0)
162 {
163 Vector2 mp = GetMousePosition();
164 spots[i].pos.x = mp.x;
165 spots[i].pos.y = screenHeight - mp.y;
166 }
167 else
168 {
169 spots[i].pos.x += spots[i].vel.x;
170 spots[i].pos.y += spots[i].vel.y;
171
172 if (spots[i].pos.x < 64) spots[i].vel.x = -spots[i].vel.x;
173 if (spots[i].pos.x > (screenWidth - 64)) spots[i].vel.x = -spots[i].vel.x;
174 if (spots[i].pos.y < 64) spots[i].vel.y = -spots[i].vel.y;
175 if (spots[i].pos.y > (screenHeight - 64)) spots[i].vel.y = -spots[i].vel.y;
176 }
177
178 SetShaderValue(shdrSpot, spots[i].posLoc, &spots[i].pos.x, SHADER_UNIFORM_VEC2);
179 }
180
181 // Draw
182 //----------------------------------------------------------------------------------
183 BeginDrawing();
184
185 ClearBackground(DARKBLUE);
186
187 // Draw stars and bobs
188 for (int n = 0; n < MAX_STARS; n++)
189 {
190 // Single pixel is just too small these days!
191 DrawRectangle((int)stars[n].pos.x, (int)stars[n].pos.y, 2, 2, WHITE);
192 }
193
194 for (int i = 0; i < 16; i++)
195 {
196 DrawTexture(texRay,
197 (int)((screenWidth/2.0f) + cos((frameCounter + i*8)/51.45f)*(screenWidth/2.2f) - 32),
198 (int)((screenHeight/2.0f) + sin((frameCounter + i*8)/17.87f)*(screenHeight/4.2f)), WHITE);
199 }
200
201 // Draw spot lights
202 BeginShaderMode(shdrSpot);
203 // Instead of a blank rectangle you could render here
204 // a render texture of the full screen used to do screen
205 // scaling (slight adjustment to shader would be required
206 // to actually pay attention to the colour!)
207 DrawRectangle(0, 0, screenWidth, screenHeight, WHITE);
208 EndShaderMode();
209
210 DrawFPS(10, 10);
211
212 DrawText("Move the mouse!", 10, 30, 20, GREEN);
213 DrawText("Pitch Black", (int)(screenWidth*0.2f), screenHeight/2, 20, GREEN);
214 DrawText("Dark", (int)(screenWidth*.66f), screenHeight/2, 20, GREEN);
215
216
217 EndDrawing();
218 //----------------------------------------------------------------------------------
219 }
220
221 // De-Initialization
222 //--------------------------------------------------------------------------------------
223 UnloadTexture(texRay);
224 UnloadShader(shdrSpot);
225
226 CloseWindow(); // Close window and OpenGL context
227 //--------------------------------------------------------------------------------------
228
229 return 0;
230}

233void ResetStar(Star *s)
234{
235 s->pos = (Vector2){ GetScreenWidth()/2.0f, GetScreenHeight()/2.0f };
236
237 do
238 {
239 s->vel.x = (float)GetRandomValue(-1000, 1000)/100.0f;
240 s->vel.y = (float)GetRandomValue(-1000, 1000)/100.0f;
241
242 } while (!(fabs(s->vel.x) + (fabs(s->vel.y) > 1)));
243
244 s->pos = Vector2Add(s->pos, Vector2Multiply(s->vel, (Vector2){ 8.0f, 8.0f }));
245}
246
247void UpdateStar(Star *s)
248{
249 s->pos = Vector2Add(s->pos, s->vel);
250
251 if ((s->pos.x < 0) || (s->pos.x > GetScreenWidth()) ||
252 (s->pos.y < 0) || (s->pos.y > GetScreenHeight()))
253 {
254 ResetStar(s);
255 }
256}
