35int main(void)
36{
37 // Initialization
38 //--------------------------------------------------------------------------------------
39 const int screenWidth = 800;
40 const int screenHeight = 450;
41
42 InitWindow(screenWidth, screenHeight, "raylib - simple shader mask");
43
44 // Define the camera to look into our 3d world
45 Camera camera = { 0 };
46 camera.position = (Vector3){ 0.0f, 1.0f, 2.0f };
47 camera.target = (Vector3){ 0.0f, 0.0f, 0.0f };
48 camera.up = (Vector3){ 0.0f, 1.0f, 0.0f };
49 camera.fovy = 45.0f;
50 camera.projection = CAMERA_PERSPECTIVE;
51
52 // Define our three models to show the shader on
53 Mesh torus = GenMeshTorus(0.3f, 1, 16, 32);
54 Model model1 = LoadModelFromMesh(torus);
55
56 Mesh cube = GenMeshCube(0.8f,0.8f,0.8f);
57 Model model2 = LoadModelFromMesh(cube);
58
59 // Generate model to be shaded just to see the gaps in the other two
60 Mesh sphere = GenMeshSphere(1, 16, 16);
61 Model model3 = LoadModelFromMesh(sphere);
62
63 // Load the shader
64 Shader shader = LoadShader(0, TextFormat("resources/shaders/glsl%i/mask.fs", GLSL_VERSION));
65
66 // Load and apply the diffuse texture (colour map)
67 Texture texDiffuse = LoadTexture("resources/plasma.png");
68 model1.materials[0].maps[MATERIAL_MAP_DIFFUSE].texture = texDiffuse;
69 model2.materials[0].maps[MATERIAL_MAP_DIFFUSE].texture = texDiffuse;
70
71 // Using MATERIAL_MAP_EMISSION as a spare slot to use for 2nd texture
72 // NOTE: Don't use MATERIAL_MAP_IRRADIANCE, MATERIAL_MAP_PREFILTER or MATERIAL_MAP_CUBEMAP as they are bound as cube maps
73 Texture texMask = LoadTexture("resources/mask.png");
74 model1.materials[0].maps[MATERIAL_MAP_EMISSION].texture = texMask;
75 model2.materials[0].maps[MATERIAL_MAP_EMISSION].texture = texMask;
76 shader.locs[SHADER_LOC_MAP_EMISSION] = GetShaderLocation(shader, "mask");
77
78 // Frame is incremented each frame to animate the shader
79 int shaderFrame = GetShaderLocation(shader, "frame");
80
81 // Apply the shader to the two models
82 model1.materials[0].shader = shader;
83 model2.materials[0].shader = shader;
84
85 int framesCounter = 0;
86 Vector3 rotation = { 0 }; // Model rotation angles
87
88 SetTargetFPS(60); // Set to run at 60 frames-per-second
89 //--------------------------------------------------------------------------------------
90
91 // Main game loop
92 while (!WindowShouldClose()) // Detect window close button or ESC key
93 {
94 // Update
95 //----------------------------------------------------------------------------------
96 UpdateCamera(&camera);
97
98 framesCounter++;
99 rotation.x += 0.01f;
100 rotation.y += 0.005f;
101 rotation.z -= 0.0025f;
102
103 // Send frames counter to shader for animation
104 SetShaderValue(shader, shaderFrame, &framesCounter, SHADER_UNIFORM_INT);
105
106 // Rotate one of the models
107 model1.transform = MatrixRotateXYZ(rotation);
108 //----------------------------------------------------------------------------------
109
110 // Draw
111 //----------------------------------------------------------------------------------
112 BeginDrawing();
113
114 ClearBackground(DARKBLUE);
115
116 BeginMode3D(camera);
117
118 DrawModel(model1, (Vector3){0.5,0,0}, 1, WHITE);
119 DrawModelEx(model2, (Vector3){-.5,0,0}, (Vector3){1,1,0}, 50, (Vector3){1,1,1}, WHITE);
120 DrawModel(model3,(Vector3){0,0,-1.5}, 1, WHITE);
121 DrawGrid(10, 1.0f); // Draw a grid
122
123 EndMode3D();
124
125 DrawRectangle(16, 698, MeasureText(TextFormat("Frame: %i", framesCounter), 20) + 8, 42, BLUE);
126 DrawText(TextFormat("Frame: %i", framesCounter), 20, 700, 20, WHITE);
127
128 DrawFPS(10, 10);
129
130 EndDrawing();
131 //----------------------------------------------------------------------------------
132 }
133
134 // De-Initialization
135 //--------------------------------------------------------------------------------------
136 UnloadModel(model1);
137 UnloadModel(model2);
138 UnloadModel(model3);
139
140 UnloadTexture(texDiffuse); // Unload default diffuse texture
141 UnloadTexture(texMask); // Unload texture mask
142
143 UnloadShader(shader); // Unload shader
144
145 CloseWindow(); // Close window and OpenGL context
146 //--------------------------------------------------------------------------------------
147
148 return 0;
149}
