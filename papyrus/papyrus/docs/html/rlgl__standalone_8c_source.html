116//------------------------------------------------------------------------------------
117// Program main entry point
118//------------------------------------------------------------------------------------
119int main(void)
120{
121 // Initialization
122 //--------------------------------------------------------------------------------------
123 const int screenWidth = 800;
124 const int screenHeight = 450;
125
126 // GLFW3 Initialization + OpenGL 3.3 Context + Extensions
127 //--------------------------------------------------------
128 glfwSetErrorCallback(ErrorCallback);
129
130 if (!glfwInit())
131 {
132 printf("GLFW3: Can not initialize GLFW\n");
133 return 1;
134 }
135 else printf("GLFW3: GLFW initialized successfully\n");
136
137 glfwWindowHint(GLFW_SAMPLES, 4);
138 glfwWindowHint(GLFW_DEPTH_BITS, 16);
139 glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3);
140 glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 3);
141 glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);
142 //glfwWindowHint(GLFW_OPENGL_DEBUG_CONTEXT, GL_TRUE);
143#if defined(__APPLE__)
144 glfwWindowHint( GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE );
145#endif
146
147 GLFWwindow *window = glfwCreateWindow(screenWidth, screenHeight, "rlgl standalone", NULL, NULL);
148
149 if (!window)
150 {
151 glfwTerminate();
152 return 2;
153 }
154 else printf("GLFW3: Window created successfully\n");
155
156 glfwSetWindowPos(window, 200, 200);
157
158 glfwSetKeyCallback(window, KeyCallback);
159
160 glfwMakeContextCurrent(window);
161 glfwSwapInterval(0);
162
163 // Load OpenGL 3.3 supported extensions
164 rlLoadExtensions(glfwGetProcAddress);
165 //--------------------------------------------------------
166
167 // Initialize OpenGL context (states and resources)
168 rlglInit(screenWidth, screenHeight);
169
170 // Initialize viewport and internal projection/modelview matrices
171 rlViewport(0, 0, screenWidth, screenHeight);
172 rlMatrixMode(RL_PROJECTION); // Switch to PROJECTION matrix
173 rlLoadIdentity(); // Reset current matrix (PROJECTION)
174 rlOrtho(0, screenWidth, screenHeight, 0, 0.0f, 1.0f); // Orthographic projection with top-left corner at (0,0)
175 rlMatrixMode(RL_MODELVIEW); // Switch back to MODELVIEW matrix
176 rlLoadIdentity(); // Reset current matrix (MODELVIEW)
177
178 rlClearColor(245, 245, 245, 255); // Define clear color
179 rlEnableDepthTest(); // Enable DEPTH_TEST for 3D
180
181 Camera camera = { 0 };
182 camera.position = (Vector3){ 5.0f, 5.0f, 5.0f }; // Camera position
183 camera.target = (Vector3){ 0.0f, 0.0f, 0.0f }; // Camera looking at point
184 camera.up = (Vector3){ 0.0f, 1.0f, 0.0f }; // Camera up vector (rotation towards target)
185 camera.fovy = 45.0f; // Camera field-of-view Y
186
187 Vector3 cubePosition = { 0.0f, 0.0f, 0.0f }; // Cube default position (center)
188 //--------------------------------------------------------------------------------------
189
190 // Main game loop
191 while (!glfwWindowShouldClose(window))
192 {
193 // Update
194 //----------------------------------------------------------------------------------
195 //camera.position.x += 0.01f;
196 //----------------------------------------------------------------------------------
197
198 // Draw
199 //----------------------------------------------------------------------------------
200 rlClearScreenBuffers(); // Clear current framebuffer
201
202 // Draw '3D' elements in the scene
203 //-----------------------------------------------
204 // Calculate projection matrix (from perspective) and view matrix from camera look at
205 Matrix matProj = MatrixPerspective((double)(camera.fovy*DEG2RAD), (double)screenWidth/(double)screenHeight, 0.01, 1000.0);
206 Matrix matView = MatrixLookAt(camera.position, camera.target, camera.up);
207
208 rlSetMatrixModelview(matView); // Set internal modelview matrix (default shader)
209 rlSetMatrixProjection(matProj); // Set internal projection matrix (default shader)
210
211 DrawCube(cubePosition, 2.0f, 2.0f, 2.0f, RED);
212 DrawCubeWires(cubePosition, 2.0f, 2.0f, 2.0f, RAYWHITE);
213 DrawGrid(10, 1.0f);
214
215 // Draw internal render batch buffers (3D data)
216 rlDrawRenderBatchActive();
217 //-----------------------------------------------
218
219 // Draw '2D' elements in the scene (GUI)
220 //-----------------------------------------------
221#define RLGL_SET_MATRIX_MANUALLY
222#if defined(RLGL_SET_MATRIX_MANUALLY)
223 matProj = MatrixOrtho(0.0, screenWidth, screenHeight, 0.0, 0.0, 1.0);
224 matView = MatrixIdentity();
225
226 rlSetMatrixModelview(matView); // Set internal modelview matrix (default shader)
227 rlSetMatrixProjection(matProj); // Set internal projection matrix (default shader)
228
229#else // Let rlgl generate and multiply matrix internally
230
231 rlMatrixMode(RL_PROJECTION); // Enable internal projection matrix
232 rlLoadIdentity(); // Reset internal projection matrix
233 rlOrtho(0.0, screenWidth, screenHeight, 0.0, 0.0, 1.0); // Recalculate internal projection matrix
234 rlMatrixMode(RL_MODELVIEW); // Enable internal modelview matrix
235 rlLoadIdentity(); // Reset internal modelview matrix
236#endif
237 DrawRectangleV((Vector2){ 10.0f, 10.0f }, (Vector2){ 780.0f, 20.0f }, DARKGRAY);
238
239 // Draw internal render batch buffers (2D data)
240 rlDrawRenderBatchActive();
241 //-----------------------------------------------
242
243 glfwSwapBuffers(window);
244 glfwPollEvents();
245 //----------------------------------------------------------------------------------
246 }
247
248 // De-Initialization
249 //--------------------------------------------------------------------------------------
250 rlglClose(); // Unload rlgl internal buffers and default shader/texture
251
252 glfwDestroyWindow(window); // Close window
253 glfwTerminate(); // Free GLFW3 resources
254 //--------------------------------------------------------------------------------------
255
256 return 0;
257}

278// Draw rectangle using rlgl OpenGL 1.1 style coding (translated to OpenGL 3.3 internally)
279static void DrawRectangleV(Vector2 position, Vector2 size, Color color)
280{
281 rlBegin(RL_TRIANGLES);
282 rlColor4ub(color.r, color.g, color.b, color.a);
283
284 rlVertex2f(position.x, position.y);
285 rlVertex2f(position.x, position.y + size.y);
286 rlVertex2f(position.x + size.x, position.y + size.y);
287
288 rlVertex2f(position.x, position.y);
289 rlVertex2f(position.x + size.x, position.y + size.y);
290 rlVertex2f(position.x + size.x, position.y);
291 rlEnd();
292}

294// Draw a grid centered at (0, 0, 0)
295static void DrawGrid(int slices, float spacing)
296{
297 int halfSlices = slices / 2;
298
299 rlBegin(RL_LINES);
300 for (int i = -halfSlices; i <= halfSlices; i++)
301 {
302 if (i == 0)
303 {
304 rlColor3f(0.5f, 0.5f, 0.5f);
305 rlColor3f(0.5f, 0.5f, 0.5f);
306 rlColor3f(0.5f, 0.5f, 0.5f);
307 rlColor3f(0.5f, 0.5f, 0.5f);
308 }
309 else
310 {
311 rlColor3f(0.75f, 0.75f, 0.75f);
312 rlColor3f(0.75f, 0.75f, 0.75f);
313 rlColor3f(0.75f, 0.75f, 0.75f);
314 rlColor3f(0.75f, 0.75f, 0.75f);
315 }
316
317 rlVertex3f((float)i*spacing, 0.0f, (float)-halfSlices*spacing);
318 rlVertex3f((float)i*spacing, 0.0f, (float)halfSlices*spacing);
319
320 rlVertex3f((float)-halfSlices*spacing, 0.0f, (float)i*spacing);
321 rlVertex3f((float)halfSlices*spacing, 0.0f, (float)i*spacing);
322 }
323 rlEnd();
324}
