23//------------------------------------------------------------------------------------
24// Program main entry point
25//------------------------------------------------------------------------------------
26int main(void)
27{
28 const int windowWidth = 800;
29 const int windowHeight = 450;
30
31 // Enable config flags for resizable window and vertical synchro
32 SetConfigFlags(FLAG_WINDOW_RESIZABLE | FLAG_VSYNC_HINT);
33 InitWindow(windowWidth, windowHeight, "raylib [core] example - window scale letterbox");
34 SetWindowMinSize(320, 240);
35
36 int gameScreenWidth = 640;
37 int gameScreenHeight = 480;
38
39 // Render texture initialization, used to hold the rendering result so we can easily resize it
40 RenderTexture2D target = LoadRenderTexture(gameScreenWidth, gameScreenHeight);
41 SetTextureFilter(target.texture, TEXTURE_FILTER_BILINEAR); // Texture scale filter to use
42
43 Color colors[10] = { 0 };
44 for (int i = 0; i < 10; i++) colors[i] = (Color){ GetRandomValue(100, 250), GetRandomValue(50, 150), GetRandomValue(10, 100), 255 };
45
46 SetTargetFPS(60); // Set our game to run at 60 frames-per-second
47 //--------------------------------------------------------------------------------------
48
49 // Main game loop
50 while (!WindowShouldClose()) // Detect window close button or ESC key
51 {
52 // Update
53 //----------------------------------------------------------------------------------
54 // Compute required framebuffer scaling
55 float scale = MIN((float)GetScreenWidth()/gameScreenWidth, (float)GetScreenHeight()/gameScreenHeight);
56
57 if (IsKeyPressed(KEY_SPACE))
58 {
59 // Recalculate random colors for the bars
60 for (int i = 0; i < 10; i++) colors[i] = (Color){ GetRandomValue(100, 250), GetRandomValue(50, 150), GetRandomValue(10, 100), 255 };
61 }
62
63 // Update virtual mouse (clamped mouse value behind game screen)
64 Vector2 mouse = GetMousePosition();
65 Vector2 virtualMouse = { 0 };
66 virtualMouse.x = (mouse.x - (GetScreenWidth() - (gameScreenWidth*scale))*0.5f)/scale;
67 virtualMouse.y = (mouse.y - (GetScreenHeight() - (gameScreenHeight*scale))*0.5f)/scale;
68 virtualMouse = Vector2Clamp(virtualMouse, (Vector2){ 0, 0 }, (Vector2){ (float)gameScreenWidth, (float)gameScreenHeight });
69
70 // Apply the same transformation as the virtual mouse to the real mouse (i.e. to work with raygui)
71 //SetMouseOffset(-(GetScreenWidth() - (gameScreenWidth*scale))*0.5f, -(GetScreenHeight() - (gameScreenHeight*scale))*0.5f);
72 //SetMouseScale(1/scale, 1/scale);
73 //----------------------------------------------------------------------------------
74
75 // Draw
76 //----------------------------------------------------------------------------------
77 // Draw everything in the render texture, note this will not be rendered on screen, yet
78 BeginTextureMode(target);
79 ClearBackground(RAYWHITE); // Clear render texture background color
80
81 for (int i = 0; i < 10; i++) DrawRectangle(0, (gameScreenHeight/10)*i, gameScreenWidth, gameScreenHeight/10, colors[i]);
82
83 DrawText("If executed inside a window,\nyou can resize the window,\nand see the screen scaling!", 10, 25, 20, WHITE);
84 DrawText(TextFormat("Default Mouse: [%i , %i]", (int)mouse.x, (int)mouse.y), 350, 25, 20, GREEN);
85 DrawText(TextFormat("Virtual Mouse: [%i , %i]", (int)virtualMouse.x, (int)virtualMouse.y), 350, 55, 20, YELLOW);
86 EndTextureMode();
87
88 BeginDrawing();
89 ClearBackground(BLACK); // Clear screen background
90
91 // Draw render texture to screen, properly scaled
92 DrawTexturePro(target.texture, (Rectangle){ 0.0f, 0.0f, (float)target.texture.width, (float)-target.texture.height },
93 (Rectangle){ (GetScreenWidth() - ((float)gameScreenWidth*scale))*0.5f, (GetScreenHeight() - ((float)gameScreenHeight*scale))*0.5f,
94 (float)gameScreenWidth*scale, (float)gameScreenHeight*scale }, (Vector2){ 0, 0 }, 0.0f, WHITE);
95 EndDrawing();
96 //--------------------------------------------------------------------------------------
97 }
98
99 // De-Initialization
100 //--------------------------------------------------------------------------------------
101 UnloadRenderTexture(target); // Unload render texture
102
103 CloseWindow(); // Close window and OpenGL context
104 //--------------------------------------------------------------------------------------
105
106 return 0;
107}
