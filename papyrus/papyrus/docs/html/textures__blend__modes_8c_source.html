23int main(void)
24{
25 // Initialization
26 //--------------------------------------------------------------------------------------
27 const int screenWidth = 800;
28 const int screenHeight = 450;
29
30 InitWindow(screenWidth, screenHeight, "raylib [textures] example - blend modes");
31
32 // NOTE: Textures MUST be loaded after Window initialization (OpenGL context is required)
33 Image bgImage = LoadImage("resources/cyberpunk_street_background.png"); // Loaded in CPU memory (RAM)
34 Texture2D bgTexture = LoadTextureFromImage(bgImage); // Image converted to texture, GPU memory (VRAM)
35
36 Image fgImage = LoadImage("resources/cyberpunk_street_foreground.png"); // Loaded in CPU memory (RAM)
37 Texture2D fgTexture = LoadTextureFromImage(fgImage); // Image converted to texture, GPU memory (VRAM)
38
39 // Once image has been converted to texture and uploaded to VRAM, it can be unloaded from RAM
40 UnloadImage(bgImage);
41 UnloadImage(fgImage);
42
43 const int blendCountMax = 4;
44 BlendMode blendMode = 0;
45
46 // Main game loop
47 while (!WindowShouldClose()) // Detect window close button or ESC key
48 {
49 // Update
50 //----------------------------------------------------------------------------------
51 if (IsKeyPressed(KEY_SPACE))
52 {
53 if (blendMode >= (blendCountMax - 1)) blendMode = 0;
54 else blendMode++;
55 }
56 //----------------------------------------------------------------------------------
57
58 // Draw
59 //----------------------------------------------------------------------------------
60 BeginDrawing();
61
62 ClearBackground(RAYWHITE);
63
64 DrawTexture(bgTexture, screenWidth/2 - bgTexture.width/2, screenHeight/2 - bgTexture.height/2, WHITE);
65
66 // Apply the blend mode and then draw the foreground texture
67 BeginBlendMode(blendMode);
68 DrawTexture(fgTexture, screenWidth/2 - fgTexture.width/2, screenHeight/2 - fgTexture.height/2, WHITE);
69 EndBlendMode();
70
71 // Draw the texts
72 DrawText("Press SPACE to change blend modes.", 310, 350, 10, GRAY);
73
74 switch (blendMode)
75 {
76 case BLEND_ALPHA: DrawText("Current: BLEND_ALPHA", (screenWidth / 2) - 60, 370, 10, GRAY); break;
77 case BLEND_ADDITIVE: DrawText("Current: BLEND_ADDITIVE", (screenWidth / 2) - 60, 370, 10, GRAY); break;
78 case BLEND_MULTIPLIED: DrawText("Current: BLEND_MULTIPLIED", (screenWidth / 2) - 60, 370, 10, GRAY); break;
79 case BLEND_ADD_COLORS: DrawText("Current: BLEND_ADD_COLORS", (screenWidth / 2) - 60, 370, 10, GRAY); break;
80 default: break;
81 }
82
83 DrawText("(c) Cyberpunk Street Environment by Luis Zuno (@ansimuz)", screenWidth - 330, screenHeight - 20, 10, GRAY);
84
85 EndDrawing();
86 //----------------------------------------------------------------------------------
87 }
88
89 // De-Initialization
90 //--------------------------------------------------------------------------------------
91 UnloadTexture(fgTexture); // Unload foreground texture
92 UnloadTexture(bgTexture); // Unload background texture
93
94 CloseWindow(); // Close window and OpenGL context
95 //--------------------------------------------------------------------------------------
96
97 return 0;
98}
