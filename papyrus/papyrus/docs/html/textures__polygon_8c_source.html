26//------------------------------------------------------------------------------------
27// Program main entry point
28//------------------------------------------------------------------------------------
29int main(void)
30{
31 // Initialization
32 //--------------------------------------------------------------------------------------
33 const int screenWidth = 800;
34 const int screenHeight = 450;
35
36 InitWindow(screenWidth, screenHeight, "raylib [textures] example - textured polygon");
37
38 // Define texture coordinates to map our texture to poly
39 Vector2 texcoords[MAX_POINTS] = {
40 (Vector2){ 0.75f, 0.0f },
41 (Vector2){ 0.25f, 0.0f },
42 (Vector2){ 0.0f, 0.5f },
43 (Vector2){ 0.0f, 0.75f },
44 (Vector2){ 0.25f, 1.0f},
45 (Vector2){ 0.375f, 0.875f},
46 (Vector2){ 0.625f, 0.875f},
47 (Vector2){ 0.75f, 1.0f},
48 (Vector2){ 1.0f, 0.75f},
49 (Vector2){ 1.0f, 0.5f},
50 (Vector2){ 0.75f, 0.0f} // Close the poly
51 };
52
53 // Define the base poly vertices from the UV's
54 // NOTE: They can be specified in any other way
55 Vector2 points[MAX_POINTS] = { 0 };
56 for (int i = 0; i < MAX_POINTS; i++)
57 {
58 points[i].x = (texcoords[i].x - 0.5f)*256.0f;
59 points[i].y = (texcoords[i].y - 0.5f)*256.0f;
60 }
61
62 // Define the vertices drawing position
63 // NOTE: Initially same as points but updated every frame
64 Vector2 positions[MAX_POINTS] = { 0 };
65 for (int i = 0; i < MAX_POINTS; i++) positions[i] = points[i];
66
67 // Load texture to be mapped to poly
68 Texture texture = LoadTexture("resources/cat.png");
69
70 float angle = 0.0f; // Rotation angle (in degrees)
71
72 SetTargetFPS(60); // Set our game to run at 60 frames-per-second
73 //--------------------------------------------------------------------------------------
74
75 // Main game loop
76 while (!WindowShouldClose()) // Detect window close button or ESC key
77 {
78 // Update
79 //----------------------------------------------------------------------------------
80 // Update points rotation with an angle transform
81 // NOTE: Base points position are not modified
82 angle++;
83 for (int i = 0; i < MAX_POINTS; i++) positions[i] = Vector2Rotate(points[i], angle*DEG2RAD);
84 //----------------------------------------------------------------------------------
85
86 // Draw
87 //----------------------------------------------------------------------------------
88 BeginDrawing();
89
90 ClearBackground(RAYWHITE);
91
92 DrawText("textured polygon", 20, 20, 20, DARKGRAY);
93
94 DrawTexturePoly(texture, (Vector2){ GetScreenWidth()/2, GetScreenHeight()/2 },
95 positions, texcoords, MAX_POINTS, WHITE);
96
97 EndDrawing();
98 //----------------------------------------------------------------------------------
99 }
100
101 // De-Initialization
102 //--------------------------------------------------------------------------------------
103 UnloadTexture(texture); // Unload texture
104
105 CloseWindow(); // Close window and OpenGL context
106 //--------------------------------------------------------------------------------------
107
108 return 0;
109}

111// Draw textured polygon, defined by vertex and texture coordinates
112// NOTE: Polygon center must have straight line path to all points
113// without crossing perimeter, points must be in anticlockwise order
114void DrawTexturePoly(Texture2D texture, Vector2 center, Vector2 *points, Vector2 *texcoords, int pointCount, Color tint)
115{
116 rlSetTexture(texture.id);
117
118 // Texturing is only supported on RL_QUADS
119 rlBegin(RL_QUADS);
120
121 rlColor4ub(tint.r, tint.g, tint.b, tint.a);
122
123 for (int i = 0; i < pointCount - 1; i++)
124 {
125 rlTexCoord2f(0.5f, 0.5f);
126 rlVertex2f(center.x, center.y);
127
128 rlTexCoord2f(texcoords[i].x, texcoords[i].y);
129 rlVertex2f(points[i].x + center.x, points[i].y + center.y);
130
131 rlTexCoord2f(texcoords[i + 1].x, texcoords[i + 1].y);
132 rlVertex2f(points[i + 1].x + center.x, points[i + 1].y + center.y);
133
134 rlTexCoord2f(texcoords[i + 1].x, texcoords[i + 1].y);
135 rlVertex2f(points[i + 1].x + center.x, points[i + 1].y + center.y);
136 }
137 rlEnd();
138
139 rlSetTexture(0);
140}
